43// Updates the given map with entries mapping the names of streams in the
44// source set to those of the corresponding streams in the destination set.
45// Corresponding streams are those with the same tag and index. Streams with
46// no match are ignored.
47//
48// For instance, given:
49// src: FOO:abc dst: FOO:bob
50// BAR:def
51// The entry 'abc' -> 'bob' is added to the map.
52absl::Status FindCorrespondingStreams(
53 std::map<std::string, std::string>* stream_map,
54 const proto_ns::RepeatedPtrField<ProtoString>& src_streams,
55 const proto_ns::RepeatedPtrField<ProtoString>& dst_streams);
