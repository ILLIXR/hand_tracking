34// AssocationCalculator<T> accepts multiple inputs of vectors of type T that can
35// be converted to Rectangle_f. The output is a vector of type T that contains
36// elements from the input vectors that don't overlap with each other. When
37// two elements overlap, the element that comes in from a later input stream
38// is kept in the output. This association operation is useful for multiple
39// instance inference pipelines in MediaPipe.
40// If an input stream is tagged with "PREV" tag, IDs of overlapping elements
41// from "PREV" input stream are propagated to the output. Elements in the "PREV"
42// input stream that don't overlap with other elements are not added to the
43// output. This stream is designed to take detections from previous timestamp,
44// e.g. output of PreviousLoopbackCalculator to provide temporal association.
45// See AssociationDetectionCalculator and AssociationNormRectCalculator for
46// example uses.
47template <typename T>
48class AssociationCalculator : public CalculatorBase {
49 public:
50 static absl::Status GetContract(CalculatorContract* cc) {
51 // Atmost one input stream can be tagged with "PREV".
52 RET_CHECK_LE(cc->Inputs().NumEntries("PREV"), 1);
53
54 if (cc->Inputs().HasTag("PREV")) {
55 RET_CHECK_GE(cc->Inputs().NumEntries(), 2);
56 }
57
58 for (CollectionItemId id = cc->Inputs().BeginId();
59 id < cc->Inputs().EndId(); ++id) {
60 cc->Inputs().Get(id).Set<std::vector<T>>();
61 }
62
63 cc->Outputs().Index(0).Set<std::vector<T>>();
64
65 return absl::OkStatus();
66 }
67
68 absl::Status Open(CalculatorContext* cc) override {
69 cc->SetOffset(TimestampDiff(0));
70
71 has_prev_input_stream_ = cc->Inputs().HasTag("PREV");
72 if (has_prev_input_stream_) {
73 prev_input_stream_id_ = cc->Inputs().GetId("PREV", 0);
74 }
75 options_ = cc->Options<::mediapipe::AssociationCalculatorOptions>();
76 ABSL_CHECK_GE(options_.min_similarity_threshold(), 0);
77
78 return absl::OkStatus();
79 }
80
81 absl::Status Process(CalculatorContext* cc) override {
82 auto get_non_overlapping_elements = GetNonOverlappingElements(cc);
83 if (!get_non_overlapping_elements.ok()) {
84 return get_non_overlapping_elements.status();
85 }
86 std::list<T> result = get_non_overlapping_elements.value();
87
88 if (has_prev_input_stream_ &&
89 !cc->Inputs().Get(prev_input_stream_id_).IsEmpty()) {
90 // Processed all regular input streams. Now compare the result list
91 // elements with those in the PREV input stream, and propagate IDs from
92 // PREV input stream as appropriate.
93 const std::vector<T>& prev_input_vec =
94 cc->Inputs()
95 .Get(prev_input_stream_id_)
96 .template Get<std::vector<T>>();
97
98 MP_RETURN_IF_ERROR(
99 PropagateIdsFromPreviousToCurrent(prev_input_vec, &result));
100 }
101
102 auto output = absl::make_unique<std::vector<T>>();
103 for (auto it = result.begin(); it != result.end(); ++it) {
104 output->push_back(*it);
105 }
106 cc->Outputs().Index(0).Add(output.release(), cc->InputTimestamp());
107
108 return absl::OkStatus();
109 }
110
111 protected:
112 ::mediapipe::AssociationCalculatorOptions options_;
113
114 bool has_prev_input_stream_;
115 CollectionItemId prev_input_stream_id_;
116
117 virtual absl::StatusOr<Rectangle_f> GetRectangle(const T& input) {
118 UNUSED(input);
119 return absl::OkStatus();
120 }
121
122 virtual std::pair<bool, int> GetId(const T& input) { UNUSED(input); return {false, -1}; }
123
124 virtual void SetId(T* input, int id) {UNUSED(input); UNUSED(id);}
125
126 private:
127 // Get a list of non-overlapping elements from all input streams, with
128 // increasing order of priority based on input stream index.
129 absl::StatusOr<std::list<T>> GetNonOverlappingElements(
130 CalculatorContext* cc) {
131 std::list<T> result;
132
133 // Initialize result with the first non-empty input vector.
134 CollectionItemId non_empty_id = cc->Inputs().BeginId();
135 for (CollectionItemId id = cc->Inputs().BeginId();
136 id < cc->Inputs().EndId(); ++id) {
137 if (id == prev_input_stream_id_ || cc->Inputs().Get(id).IsEmpty()) {
138 continue;
139 }
140 const std::vector<T>& input_vec =
141 cc->Inputs().Get(id).Get<std::vector<T>>();
142 if (!input_vec.empty()) {
143 non_empty_id = id;
144 result.push_back(input_vec[0]);
145 for (int j = 1; j < (int)input_vec.size(); ++j) {
146 MP_RETURN_IF_ERROR(AddElementToList(input_vec[j], &result));
147 }
148 break;
149 }
150 }
151
152 // Compare remaining input vectors with the non-empty result vector,
153 // remove lower-priority overlapping elements from the result vector and
154 // had corresponding higher-priority elements as necessary.
155 for (CollectionItemId id = non_empty_id + 1; id < cc->Inputs().EndId();
156 ++id) {
157 if (id == prev_input_stream_id_ || cc->Inputs().Get(id).IsEmpty()) {
158 continue;
159 }
160 const std::vector<T>& input_vec =
161 cc->Inputs().Get(id).Get<std::vector<T>>();
162
163 for (int vi = 0; vi < (int)input_vec.size(); ++vi) {
164 MP_RETURN_IF_ERROR(AddElementToList(input_vec[vi], &result));
165 }
166 }
167
168 return result;
169 }
170
171 absl::Status AddElementToList(T element, std::list<T>* current) {
172 // Compare this element with elements of the input collection. If this
173 // element has high overlap with elements of the collection, remove
174 // those elements from the collection and add this element.
175 MP_ASSIGN_OR_RETURN(auto cur_rect, GetRectangle(element));
176
177 bool change_id = false;
178 int new_elem_id = -1;
179
180 for (auto uit = current->begin(); uit != current->end();) {
181 MP_ASSIGN_OR_RETURN(auto prev_rect, GetRectangle(*uit));
182 if (CalculateIou(cur_rect, prev_rect) >
183 options_.min_similarity_threshold()) {
184 std::pair<bool, int> prev_id = GetId(*uit);
185 // If prev_id.first is false when some element doesn't have an ID,
186 // change_id and new_elem_id will not be updated.
187 if (prev_id.first) {
188 change_id = prev_id.first;
189 new_elem_id = prev_id.second;
190 }
191 uit = current->erase(uit);
192 } else {
193 ++uit;
194 }
195 }
196
197 if (change_id) {
198 SetId(&element, new_elem_id);
199 }
200 current->push_back(element);
201
202 return absl::OkStatus();
203 }
204
205 // Compare elements of the current list with elements in from the collection
206 // of elements from the previous input stream, and propagate IDs from the
207 // previous input stream as appropriate.
208 absl::Status PropagateIdsFromPreviousToCurrent(
209 const std::vector<T>& prev_input_vec, std::list<T>* current) {
210 for (auto vit = current->begin(); vit != current->end(); ++vit) {
211 auto get_cur_rectangle = GetRectangle(*vit);
212 if (!get_cur_rectangle.ok()) {
213 return get_cur_rectangle.status();
214 }
215 const Rectangle_f& cur_rect = get_cur_rectangle.value();
216
217 bool change_id = false;
218 int id_for_vi = -1;
219
220 for (int ui = 0; ui < (int)prev_input_vec.size(); ++ui) {
221 auto get_prev_rectangle = GetRectangle(prev_input_vec[ui]);
222 if (!get_prev_rectangle.ok()) {
223 return get_prev_rectangle.status();
224 }
225 const Rectangle_f& prev_rect = get_prev_rectangle.value();
226
227 if (CalculateIou(cur_rect, prev_rect) >
228 options_.min_similarity_threshold()) {
229 std::pair<bool, int> prev_id = GetId(prev_input_vec[ui]);
230 // If prev_id.first is false when some element doesn't have an ID,
231 // change_id and id_for_vi will not be updated.
232 if (prev_id.first) {
233 change_id = prev_id.first;
234 id_for_vi = prev_id.second;
235 }
236 }
237 }
238
239 if (change_id) {
240 T element = *vit;
241 SetId(&element, id_for_vi);
242 *vit = element;
243 }
244 }
245 return absl::OkStatus();
246 }
247};
