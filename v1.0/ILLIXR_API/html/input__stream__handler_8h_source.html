43// Indicates the operation the node is ready for.
44enum class NodeReadiness {
45 // The node is not ready.
46 kNotReady,
47 // The node is ready and we should run Process().
48 kReadyForProcess,
49 // The node is ready and we should run Close().
50 kReadyForClose,
51};

53// Abstract base class for input stream handlers.
54//
55// The input stream handler is invoked every time a Packet is enqueued or the
56// next timestamp bound is set in any of the input streams and actually decides
57// whether a calculator node's Process() should be called. The input stream
58// handler owns and manages the InputStreamManager objects. The operations
59// performed by OutputStreamHandler on an InputStreamManager object need to go
60// through its input stream handler.
61//
62// Derived classes are required to implement the following member functions:
63//
64// NodeReadiness GetNodeReadiness(Timestamp* min_stream_timestamp);
65// void FillInputSet(Timestamp input_timestamp,
66// InputStreamShardSet* input_set);
67//
68class InputStreamHandler {
69 public:
70 InputStreamHandler(std::shared_ptr<tool::TagMap> tag_map,
71 CalculatorContextManager* calculator_context_manager,
72 const MediaPipeOptions& options,
73 bool calculator_run_in_parallel)
74 : input_stream_managers_(std::move(tag_map)),
75 calculator_context_manager_(calculator_context_manager),
76 options_(options),
77 calculator_run_in_parallel_(calculator_run_in_parallel) {}
78
79 virtual ~InputStreamHandler() = default;

153 // Keeps scheduling new invocations until 1) the node is not ready or 2) the
154 // max number of invocations that are allowed to be scheduled is reached.
155 // Returns true if at least one invocation has been scheduled.
156 // The latest minimum timestamp bound of the input streams is returned in
157 // *input_bound if the latest readiness of the node is kNotReady when the
158 // function returns. During batching, this value will be equal to the
159 // timestamp of the first set of inputs in the batch. In other cases,
160 // Timestamp::Unset() is returned.
161 // This method can only be invoked in the schedule phase.
162 bool ScheduleInvocations(int max_allowance, Timestamp* input_bound);

164 // Finalizes the input set before the calculator node's Process() is

168 // Returns the number of input stream headers (excluding headers of back

174 // When true, Calculator::Process is called for any increase in the
175 // timestamp bound, whether or not any packets are available.
176 // Calculator::Process is called when the minimum timestamp bound
177 // increases for any synchronized set of input streams.
178 // DefaultInputStreamHandler groups all input streams into a single set.
179 // ImmediateInputStreamHandler treats each input stream as a separate set.
180 void SetProcessTimestampBounds(bool process_ts) {
181 process_timestamps_ = process_ts;
182 }
183
184 // When true, Calculator::Process is called for every input timestamp bound.
185 bool ProcessTimestampBounds() { return process_timestamps_; }

190 // A helper class to build input packet sets for a certain set of streams.
191 //
192 // ReadyForProcess requires all of the streams to be fully determined
193 // at the same input-timestamp.
194 // This is the readiness policy for all streams in DefaultInputStreamHandler.
195 // It is also the policy for each sync-set in SyncSetInputStreamHandler.
196 // It is also the policy for each input-stream in ImmediateInputStreamHandler.
197 //
198 // If ProcessTimestampBounds() is set, then a fully determined input timestamp
199 // with only empty input packets will qualify as ReadyForProcess.
200 class SyncSet {
201 public:
202 // Creates a SyncSet for a certain set of streams, |stream_ids|.
203 SyncSet(InputStreamHandler* input_stream_handler,
204 std::vector<CollectionItemId> stream_ids);
205
206 // Reinitializes this SyncSet before each CalculatorGraph run.
207 void PrepareForRun();
208

210 NodeReadiness GetReadiness(Timestamp* min_stream_timestamp);
211
212 // Returns the latest timestamp returned for processing.
213 Timestamp LastProcessed() const;
214
215 // The earliest available packet timestamp, or Timestamp::Done.
216 Timestamp MinPacketTimestamp() const;
217
218 // Moves packets from all input streams to the input_set.
219 void FillInputSet(Timestamp input_timestamp,
220 InputStreamShardSet* input_set);
221
222 // Copies timestamp bounds from all input streams to the input_set.
223 void FillInputBounds(InputStreamShardSet* input_set);
224
225 private:
226 InputStreamHandler* input_stream_handler_;
227 std::vector<CollectionItemId> stream_ids_;
228 Timestamp last_processed_ts_ = Timestamp::Unset();
229 };

244 // Accesses InputStreamShard's private method to add packets. All
245 // InputStreamHandler subclasses are only able to add packets to shards
246 // through this method.
247 // TODO: Renames this method to "SetStreamContents" when
248 // InputStreamShard is renamed.
249 void AddPacketToShard(InputStreamShard* shard, Packet&& value, bool is_done) {
250 shard->AddPacket(std::move(value), is_done);
251 }
252
253 // Returns the operation the calculator node is ready for.
254 // Specifically:
255 // - NodeReadiness::kNotReady if the node's Process() or Close() cannot be
256 // called yet. The minimum timestamp bound of the input streams is
257 // returned in *min_stream_timestamp.
258 // - NodeReadiness::kReadyForProcess if the node is not waiting for a packet
259 // on any input stream and is thus "ready" for its Process() to be
260 // called. For a source node, Timestamp::Done() is returned in
261 // *min_stream_timestamp. For a non-source node, the input timestamp for
262 // the Process() call is returned in *min_stream_timestamp.
263 // - NodeReadiness::kReadyForClose if the node's Close() can be called.
264 // Timestamp::Done() is returned in *min_stream_timestamp.
265 virtual NodeReadiness GetNodeReadiness(Timestamp* min_stream_timestamp) = 0;
266
267 // Moves input packets from the input streams into the input set for the given
268 // input timestamp.
269 virtual void FillInputSet(Timestamp input_timestamp,
270 InputStreamShardSet* input_set) = 0;
271
272 // Collection of InputStreamManager objects.
273 InputStreamManagerSet input_stream_managers_;
274 // A pointer to the calculator context manager of the calculator node.
275 CalculatorContextManager* const calculator_context_manager_;
276 MediaPipeOptions options_;
277 const bool calculator_run_in_parallel_;
278 // Indicates if a calculator context has been prepared for Close().
279 bool prepared_context_for_close_;
280
281 // Note: You must not hold any locks while invoking any of these callbacks.
282 //
283 // A callback to notify the observer of the changes on the packet queue or
284 // the timestamp bound.
285 std::function<void()> notification_;
286 // A callback to schedule the node with the prepared calculator context.
287 std::function<void(CalculatorContext*)> schedule_callback_;
288 std::function<void(absl::Status)> error_callback_;
289
290 private:
291 // Indicates when to fill the input set. If true, every input set will be
292 // prepared in FinalizeInputSet(). Otherwise, the input sets will be filled
293 // in ScheduleInvocations() in the scheduling phase.
294 // The variable is set to false by default. A subclass should set it to true
295 // with SetLatePreparation(true) in the constructor if the input sets need to
296 // be filled in ProcessNode().
297 bool late_preparation_ = false;
298
299 // Determines how many sets of input packets are collected before a
300 // CalculatorNode is scheduled.
301 int batch_size_ = 1;
302
303 // When true, any increase in timestamp bound invokes Calculator::Process.
304 bool process_timestamps_ = false;
305
306 // A callback to notify the observer when all the input stream headers
307 // (excluding headers of back edges) become available.
308 std::function<void()> headers_ready_callback_;
309
310 std::atomic<int> unset_header_count_{0};
311};
