37template <class Item>
38class ReusablePool : public std::enable_shared_from_this<ReusablePool<Item>> {
39 public:
40 using ItemFactory =
41 absl::AnyInvocable<absl::StatusOr<std::unique_ptr<Item>>() const>;
42
43 // Creates a pool. This pool will manage buffers of the specified dimensions,
44 // and will keep keep_count buffers around for reuse.
45 // We enforce creation as a shared_ptr so that we can use a weak reference in
46 // the buffers' deleters.
47 static std::shared_ptr<ReusablePool<Item>> Create(
48 ItemFactory item_factory, const MultiPoolOptions& options) {
49 return std::shared_ptr<ReusablePool<Item>>(
50 new ReusablePool<Item>(std::move(item_factory), options));
51 }
52
53 // Obtains a buffer. May either be reused or created anew.
54 // A GlContext must be current when this is called.
55 absl::StatusOr<std::shared_ptr<Item>> GetBuffer();
56
57 // This method is meant for testing.
58 std::pair<int, int> GetInUseAndAvailableCounts();
59
60 protected:
61 ReusablePool(ItemFactory item_factory, const MultiPoolOptions& options)
62 : item_factory_(std::move(item_factory)),
63 keep_count_(options.keep_count) {}
64
65 private:
66 // Return a buffer to the pool.
67 void Return(std::unique_ptr<Item> buf);
68
69 // If the total number of buffers is greater than keep_count, destroys any
70 // surplus buffers that are no longer in use.
71 void TrimAvailable(std::vector<std::unique_ptr<Item>>* trimmed)
72 ABSL_EXCLUSIVE_LOCKS_REQUIRED(mutex_);
73
74 const ItemFactory item_factory_;
75 const int keep_count_;
76
77 absl::Mutex mutex_;
78 int in_use_count_ ABSL_GUARDED_BY(mutex_) = 0;
79 std::vector<std::unique_ptr<Item>> available_ ABSL_GUARDED_BY(mutex_);
80};
81
82template <class Item>
83inline absl::StatusOr<std::shared_ptr<Item>> ReusablePool<Item>::GetBuffer() {
84 std::unique_ptr<Item> buffer;
85 bool reuse = false;
86
87 {
88 absl::MutexLock lock(&mutex_);
89 if (available_.empty()) {
90 MP_ASSIGN_OR_RETURN(buffer, item_factory_());
91 RET_CHECK_NE(buffer, nullptr) << "Failed to create buffer";
92 } else {
93 buffer = std::move(available_.back());
94 available_.pop_back();
95 reuse = true;
96 }
97
98 ++in_use_count_;
99 }
100
101 // This needs to wait on consumer sync points, therefore it should not be
102 // done while holding the mutex.
103 if (reuse) {
104 buffer->Reuse();
105 }
106
107 // Return a shared_ptr with a custom deleter that adds the buffer back
108 // to our available list.
109 std::weak_ptr<ReusablePool<Item>> weak_pool(this->shared_from_this());
110 return std::shared_ptr<Item>(buffer.release(), [weak_pool](Item* buf) {
111 auto pool = weak_pool.lock();
112 if (pool) {
113 pool->Return(absl::WrapUnique(buf));
114 } else {
115 delete buf;
116 }
117 });
118}
119
120template <class Item>
121inline std::pair<int, int> ReusablePool<Item>::GetInUseAndAvailableCounts() {
122 absl::MutexLock lock(&mutex_);
123 return {in_use_count_, available_.size()};
124}
125
126template <class Item>
127void ReusablePool<Item>::Return(std::unique_ptr<Item> buf) {
128 std::vector<std::unique_ptr<Item>> trimmed;
129 {
130 absl::MutexLock lock(&mutex_);
131 --in_use_count_;
132 available_.emplace_back(std::move(buf));
133 TrimAvailable(&trimmed);
134 }
135 // The trimmed buffers will be released without holding the lock.
136}
137
138template <class Item>
139void ReusablePool<Item>::TrimAvailable(
140 std::vector<std::unique_ptr<Item>>* trimmed) {
141 int keep = std::max(keep_count_ - in_use_count_, 0);
142 if (available_.size() > keep) {
143 ABSL_LOG_FIRST_N(WARNING, 1)
144 << "Maximum number of pooled buffers reached. Consider increasing "
145 "MultiPoolOptions.keep_count";
146 auto trim_it = std::next(available_.begin(), keep);
147 if (trimmed) {
148 std::move(trim_it, available_.end(), std::back_inserter(*trimmed));
149 }
150 available_.erase(trim_it, available_.end());
151 }
152}
