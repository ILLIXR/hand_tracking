16// ::mediapipe::formats::MatView(&frame);

19// frame.CopyPixelData(format, width, height, raw_data_ptr,
20// ImageFrame::kDefaultAlignmentBoundary);

22// Convert an RGB ImageFrame (rgb_frame) to Grayscale:
23// ImageFrame gray_frame(ImageFormat::GRAY8, rgb_frame.Width(),
24// rgb_frame.Height());
25// cv::Mat rgb_frame_mat = ::mediapipe::formats::MatView(&rgb_frame);
26// cv::Mat gray_frame_mat = ::mediapipe::formats::MatView(&gray_frame);
27// cv::cvtColor(rgb_frame_mat, gray_frame_mat, CV_RGB2GRAY);

30// ImageFrame small_image(ImageFormat::GRAY8, 10, 10);
31// cv::Mat destination = ::mediapipe::formats::MatView(&small_image);

53// A container for storing an image or a video frame, in one of several
54// formats. Pixels are encoded row-major in an interleaved fashion.
55//
56// Formats supported by ImageFrame are listed in the ImageFormat proto.
57// It is the intention of ImageFormat to specify both the data format
58// and the colorspace used. For example GRAY8 and GRAY16 both use the
59// same colorspace but have different formats. Although it would be
60// possible to keep HSV, linearRGB, or BGR values inside an ImageFrame
61// (with format SRGB) this is an abuse of the class. If you need a new
62// format, please add one to ImageFormat::Format.
63//
64// Do not assume that the pixel data is stored contiguously. It may be
65// stored with row padding for alignment purposes.
66class ImageFrame {
67 public:
68 typedef std::function<void(uint8_t*)> Deleter;
69
70 // This class offers a few standard delete functions and retains
71 // compatibility with the previous API.
72 class PixelDataDeleter {
73 public:
74 static const Deleter kArrayDelete;
75 static const Deleter kFree;
76 static const Deleter kAlignedFree;
77 static const Deleter kNone;
78 };
79
80 // Use a default alignment boundary of 16 because Intel SSE2 instructions may
81 // incur performance penalty when accessing data not aligned on a 16-byte
82 // boundary. FFmpeg requires at least this level of alignment.
83 static const uint32_t kDefaultAlignmentBoundary = 16;
84
85 // If the pixel data of an ImageFrame will be passed to an OpenGL function
86 // such as glTexImage2D() or glReadPixels(), use a four-byte alignment
87 // boundary because that is the initial value of the OpenGL GL_PACK_ALIGNMENT
88 // and GL_UNPACK_ALIGNMENT parameters.
89 static const uint32_t kGlDefaultAlignmentBoundary = 4;
90
91 // Returns number of channels for the given image format.
92 static int NumberOfChannelsForFormat(ImageFormat::Format format);
93 // Returns the size of each channel in bytes for the given image format.
94 static int ChannelSizeForFormat(ImageFormat::Format format);
95 ABSL_DEPRECATED("Use ChannelSizeForFormat() instead")
96 static int ByteDepthForFormat(ImageFormat::Format format);
97
98 ImageFrame(const ImageFrame&) = delete;
99 ImageFrame& operator=(const ImageFrame&) = delete;
100 // Creates an empty ImageFrame. It will need to be initialized by some other
101 // means.
102 ImageFrame();
103
104 // Allocate a frame of the appropriate size. Does not zero it out.
105 // Each row will be aligned to alignment_boundary. alignment_boundary
106 // must be a power of 2 (the number 1 is valid, and means the data will
107 // be stored contiguously).
108 ImageFrame(ImageFormat::Format format, int width, int height,
109 uint32_t alignment_boundary);
110 // Same as above, but use kDefaultAlignmentBoundary for alignment_boundary.
111 ImageFrame(ImageFormat::Format format, int width, int height);
112
113 // Acquires ownership of pixel_data. Sets the deletion method
114 // to use on pixel_data with deletion_method (which defaults
115 // to using delete[]). pixel_data must have been allocated of
116 // size at least width_step*height and width_step must be at least
117 // width*num_channels*depth. Both width_step and depth are in units
118 // of bytes.
119 ImageFrame(ImageFormat::Format format, int width, int height, int width_step,
120 uint8_t* pixel_data,
121 Deleter deleter = std::default_delete<uint8_t[]>());
122
123 ImageFrame(ImageFrame&& move_from);
124 ImageFrame& operator=(ImageFrame&& move_from);
125
126 // Returns true if the ImageFrame is unallocated.
127 bool IsEmpty() const { return pixel_data_ == nullptr; }
128
129 // Set the entire frame allocation to zero, including alignment
130 // padding areas.
131 void SetToZero();
132 // Set the padding bytes at the end of each row (that are used for
133 // alignment) to deterministic values. This function should be called
134 // to get deterministic behavior from functions that read the padding
135 // areas (generally as part of highly optimized operations such as
136 // those in ffmpeg).
137 void SetAlignmentPaddingAreas();
138
139 // Returns true if the data is stored contiguously (without any
140 // alignment padding areas).
141 bool IsContiguous() const;
142
143 // Returns true if each row of the data is aligned to
144 // alignment_boundary. If IsAligned(16) is true then so are
145 // IsAligned(8), IsAligned(4), IsAligned(2), and IsAligned(1).
146 // alignment_boundary must be 1 or a power of 2.
147 bool IsAligned(uint32_t alignment_boundary) const;
148
149 // Returns the image / video format.
150 ImageFormat::Format Format() const { return format_; }
151 // Returns the width of the image in pixels.
152 int Width() const { return width_; }
153 // Returns the height of the image in pixels.
154 int Height() const { return height_; }
155 // Returns the number of channels.
156 int NumberOfChannels() const;
157 // Returns the size of each channel in bytes.
158 int ChannelSize() const;
159 ABSL_DEPRECATED("Use ChannelSize() instead")
160 int ByteDepth() const;
161
162 // Returns the byte offset between a pixel value and the same pixel
163 // and channel in the next row. Notice, that for alignment reasons,
164 // there may be unused padding bytes at the end of each row
165 // (WidthStep() - Width()*NumberOfChannels*ChannelSize() will give the
166 // number of unused bytes).
167 int WidthStep() const { return width_step_; }
168
169 // Reset the current image frame and copy the data from image_frame into
170 // this image frame. The alignment_boundary must be given (and won't
171 // necessarily match the alignment_boundary of the input image_frame).
172 void CopyFrom(const ImageFrame& image_frame, uint32_t alignment_boundary);
173
174 // Get a mutable pointer to the underlying image data. The ImageFrame
175 // retains ownership.
176 uint8_t* MutablePixelData() { return pixel_data_.get(); }
177 // Get a const pointer to the underlying image data.
178 const uint8_t* PixelData() const { return pixel_data_.get(); }
179
180 // Returns the total size of the pixel data.
181 int PixelDataSize() const { return Height() * WidthStep(); }
182 // Returns the total size the pixel data would take if it was stored
183 // contiguously (which may not be the case).
184 int PixelDataSizeStoredContiguously() const {
185 return Width() * Height() * ChannelSize() * NumberOfChannels();
186 }
187
188 // Initializes ImageFrame from pixel data without copying.
189 // ImageFrame takes ownership of pixel_data. See the Constructor
190 // with the same arguments for details.
191 void AdoptPixelData(ImageFormat::Format format, int width, int height,
192 int width_step, uint8_t* pixel_data,
193 Deleter deleter = std::default_delete<uint8_t[]>());
194
195 // Resets the ImageFrame and makes it a copy of the provided pixel
196 // data, which is assumed to be stored contiguously. The ImageFrame
197 // will use the given alignment_boundary.
198 void CopyPixelData(ImageFormat::Format format, int width, int height,
199 const uint8_t* pixel_data, uint32_t alignment_boundary);
200
201 // Resets the ImageFrame and makes it a copy of the provided pixel
202 // data, with given width_step. The ImageFrame
203 // will use the given alignment_boundary.
204 void CopyPixelData(ImageFormat::Format format, int width, int height,
205 int width_step, const uint8_t* pixel_data,
206 uint32_t alignment_boundary);
207
208 // Allocates a frame of the specified format, width, height, and alignment,
209 // without clearing any current pixel data. See the constructor with the same
210 // argument list.
211 void Reset(ImageFormat::Format format, int width, int height,
212 uint32_t alignment_boundary);
213
214 // Relinquishes ownership of the pixel data. Notice that the unique_ptr
215 // uses a non-standard deleter.
216 std::unique_ptr<uint8_t[], Deleter> Release();
217
218 // Copy the 8-bit ImageFrame into a contiguous, pre-allocated buffer. Note
219 // that ImageFrame does not necessarily store its data contiguously (i.e. do
220 // not use copy_n to move image data).
221 void CopyToBuffer(uint8_t* buffer, int buffer_size) const;
222
223 // A version of CopyToBuffer for 16-bit pixel data. Note that buffer_size
224 // stores the number of 16-bit elements in the buffer, not the number of
225 // bytes.
226 void CopyToBuffer(uint16_t* buffer, int buffer_size) const;
227
228 // A version of CopyToBuffer for float pixel data. Note that buffer_size
229 // stores the number of float elements in the buffer, not the number of
230 // bytes.
231 void CopyToBuffer(float* buffer, int buffer_size) const;
232
233 // Returns an error message which prints out the format encountered.
234 static std::string InvalidFormatString(ImageFormat::Format format);
235
236 private:
237 // Returns true if alignment_number is 1 or a power of 2.
238 static bool IsValidAlignmentNumber(uint32_t alignment_boundary);
239
240 // The internal implementation of copying data from the provided pixel data.
241 // If width_step is 0, then calculates width_step assuming no padding.
242 void InternalCopyFrom(int width, int height, int width_step, int channel_size,
243 const uint8_t* pixel_data);
244
245 // The internal implementation of copying data to the provided buffer.
246 // If width_step is 0, then calculates width_step assuming no padding.
247 void InternalCopyToBuffer(int width_step, char* buffer) const;
248
249 ImageFormat::Format format_;
250 int width_;
251 int height_;
252 int width_step_;
253 std::unique_ptr<uint8_t[], Deleter> pixel_data_;
254};
