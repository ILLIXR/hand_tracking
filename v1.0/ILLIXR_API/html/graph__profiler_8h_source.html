42struct PacketId {
43 // Stream name, excluding TAG if available.
44 std::string stream_name;
45 // Timestamp of the packet.
46 int64_t timestamp_usec;
47
48 bool operator==(const PacketId& other) const {
49 return (stream_name == other.stream_name) &&
50 (timestamp_usec == other.timestamp_usec);
51 }
52};

54struct PacketInfo {
55 // Number of remained consumer of this packet.
56 // This is used to decide if this PacketInfo should be discarded.
57 int64_t remaining_consumer_count;
58 // Packet production time based on profiler's clock.
59 int64_t production_time_usec;
60 // The time when the Process(), that generated the corresponding source
61 // packet, was started.
62 int64_t source_process_start_usec;
63
64 // For testing.
65 bool operator==(const PacketInfo& other) const {
66 return (remaining_consumer_count == other.remaining_consumer_count) &&
67 (production_time_usec == other.production_time_usec) &&
68 (source_process_start_usec == other.source_process_start_usec);
69 }
70};

78// GraphProfiler keeps track of the following in microseconds based on the
79// profiler clock, for each calculator
80// - Open(), Process(), and Close() runtime.
81// - Input stream latency: Time from when a packet was produced to when it was
82// consumed by the calculator.
83// - Process input latency: How long it took a packet to travel from start of
84// the graph (source nodes) to reach the Calculator.
85// - Process input latency: Process input latency + process runtime for a
86// packet.
87//
88// The profiler can be configured in the graph definition:
89// profiler_config {
90// histogram_interval_size_usec : 2000000
91// num_histogram_intervals : 5
92// enable_profiler: true
93// }
94//
95// Because the graph definition affects the stream profiling and the profiler is
96// singleton, the profiler can not be used with more than one graph. Thus the
97// profiler disables itself and returns an empty stub if Initialize() is called
98// more than once.
99//
100// The profiler uses the synchronized monotonic clock by default.
101// The client can overwrite this by calling SetClock().
102class GraphProfiler : public std::enable_shared_from_this<ProfilingContext> {

171 // Convenience temporary object to record scoped entry and exit.
172 // Gets start_time_usec_ on construction and records process runtime on
173 // destruction. The |calculator_context| and |profiler| must not be null.
174 class Scope {
175 public:
176 // Constructs a scope.
177 //
178 // REQUIRES: `calculator_context` and `profiler` are not null, and must both
179 // outlive this instance.
180 inline explicit Scope(GraphTrace::EventType event_type,
181 CalculatorContext* calculator_context,
182 GraphProfiler* profiler)
183 : calculator_method_(event_type),
184 calculator_context_(*calculator_context),
185 profiler_(profiler) {
186 start_time_usec_ = profiler_->TimeNowUsec();
187 if (profiler_->is_tracing_) {
188 absl::Time time_now = absl::FromUnixMicros(start_time_usec_);
189 profiler_->packet_tracer_->LogInputEvents(
190 calculator_method_, &calculator_context_, time_now);
191 }
192 }
193
194 inline ~Scope() {
195 int64_t end_time_usec;
196 if (profiler_->is_profiling_ || profiler_->is_tracing_) {
197 end_time_usec = profiler_->TimeNowUsec();
198 }
199 if (profiler_->is_profiling_) {
200 int64_t end_time_usec = profiler_->TimeNowUsec();
201 switch (calculator_method_) {
202 case GraphTrace::OPEN:
203 profiler_->SetOpenRuntime(calculator_context_, start_time_usec_,
204 end_time_usec);
205 break;
206
207 case GraphTrace::PROCESS:
208 profiler_->AddProcessSample(calculator_context_, start_time_usec_,
209 end_time_usec);
210 break;
211
212 case GraphTrace::CLOSE:
213 profiler_->SetCloseRuntime(calculator_context_, start_time_usec_,
214 end_time_usec);
215 break;
216 default:
217 break;
218 }
219 }
220 if (profiler_->is_tracing_) {
221 absl::Time time_now = absl::FromUnixMicros(end_time_usec);
222 profiler_->packet_tracer_->LogOutputEvents(
223 calculator_method_, &calculator_context_, time_now);
224 }
225 }
226
227 private:
228 const GraphTrace::EventType calculator_method_;
229 const CalculatorContext& calculator_context_;
230 GraphProfiler* profiler_;
231 int64_t start_time_usec_;
232 };

386// GlContextProfiler keeps track of all timestamp queries within a specific
387// GlContext object. When created, the GlContextProfiler must be initialized
388// before marking timestamps. Finally, when GlContext is no longer interested
389// in marking timestamps or is about to be destroyed, Finish() must be called
390// to complete all pending time queries and detach the timer from the GlContext.
391// Note that the GlContextProfiler must be created and initialized within a
392// valid GlContext object.
393#if !MEDIAPIPE_DISABLE_GPU_PROFILER
394class GlContextProfiler {
395 public:
396 explicit GlContextProfiler(
397 std::shared_ptr<ProfilingContext> profiling_context)
398 : profiling_context_(profiling_context) {}
399
400 // Not copyable or movable.
401 GlContextProfiler(const GlContextProfiler&) = delete;
402 GlContextProfiler& operator=(const GlContextProfiler&) = delete;
403
404 // Add a GlTimingInfo object to the collection of pending timestamp queries
405 // associated with a specific graph node_id, packet input_timestamp and mark
406 // if it is a start or stop event. When a stop event is marked, this function
407 // blocks on the corresponding start event to complete.
408 void MarkTimestamp(int node_id, Timestamp input_timestamp, bool is_finish);
409
410 // Complete all pending timing queries and detach the timer from the
411 // GlContext.
412 void LogAllTimestamps();
413
414 private:
415 // Store GlTimeQuery and the corresponding TraceEvent object that should be
416 // populated when the query completes together.
417 struct GlTimingInfo {
418 GlTimeQuery time_query;
419 TraceEvent trace_event;
420 };
421
422 // Setup the timer for marking GPU timestamps. If successful in setup, return
423 // true otherwise return false to indicate that timing measurment is not
424 // supported.
425 bool Initialize();
426
427 absl::Time TimeNow();
428
429 // Calibrate the GPU timer w.r.t. the CPU clock. If calibration is fails,
430 // timing_measurement_supported_ is set to false.
431 void CalibrateTimer(bool recalibrate);
432
433 // Log a TraceEvent object to represent if the GPU calibration period has
434 // started or just ended.
435 void LogCalibrationEvent(bool started, absl::Time time);
436
437 // Log TraceEvent objects for completed time queries. If the parameter wait is
438 // set to true, wait for all time queries to complete before returning.
439 void RetireReadyGlTimings(bool wait = false);
440
441 // Get the TraceEvent object containing the timestamp recorded by the GPU if
442 // the provided query was fulfilled. If it is still pending and wait is false,
443 // return absl::nullopt.
444 absl::optional<TraceEvent> GetTimeFromQuery(
445 std::unique_ptr<GlTimingInfo>& query, bool wait);
446
447 std::shared_ptr<ProfilingContext> profiling_context_;
448 GlSimpleTimer gl_timer_;
449 bool checked_timing_supported_ = false;
450 bool timing_measurement_supported_ = false;
451 std::deque<std::unique_ptr<GlTimingInfo>> pending_gl_times_;
452 std::unique_ptr<GlTimingInfo> gl_start_query_;
453};
