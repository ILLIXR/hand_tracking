15// Functions for adding Calculators that dump data from a Graph.
16//
17// Specifically this is accomplished by adding a CallbackCalculator to
18// the CalculatorGraphConfig and adding a corresponding InputSidePacket
19// to a CalculatorGraph such that data which is sent on a stream will
20// be captured in the desired way. These functions are meant to isolate
21// clients from such messy details.
22//
23// Although these functions are basically manipulations on a
24// CalculatorGraphConfig they are not placed in tool/graph.h since they
25// also depend on CalculatorGraph and having them in tool/graph.h would
26// introduce a circular dependency.

58// Add a CallbackCalculator to the config and an InputSidePacket to
59// the uninitialized_graph such that the packets from stream_name
60// will get dumped to dumped_data when the CalculatorGraph is Run.
61// The provided graph must have been constructed but not initialized
62// (this allows an InputSidePacket to be added to the CalculatorGraph,
63// and also allows the CalculatorGraphConfig to still be changed before
64// it is used for initialization). Any number of these functions can
65// be called on graph.
66//
67// NOTE: `dumped_data` (vector sink you pass) must outlive graph initialized
68// with the `config` you add the sink to.
69//
70// Example usage:
71// CalculatorGraphConfig config = tool::ParseGraphFromFileOrDie("config.txt");
72// std::vector<Packet> packet_dump;
73// tool::AddVectorSink("output_samples", &config, &packet_dump);
74// // Call tool::AddVectorSink() more times if you wish. Note that each stream
75// // needs to get its own packet vector.
76// CalculatorGraph graph;
77// ABSL_CHECK_OK(graph.Initialize(config));
78// // Set other input side packets.
79// ABSL_CHECK_OK(graph.Run());
80// for (const Packet& packet : packet_dump) {
81// // Do something.
82// }
83void AddVectorSink(const std::string& stream_name, //
84 CalculatorGraphConfig* config, //
85 std::vector<Packet>* dumped_data);

151// TODO Move CallbackCalculator and CallbackWithHeaderCalculator to
152// a separate library, and the library will be alwayslink. Then, the "sink"
153// cc_library can depend on that library, and it does not need to be alwayslink.
154//
155// CallbackCalculator calls a user settable callback on every incoming
156// packet. It must have a single input stream and no output streams.
157// A single input side packet must be given which contains a std::function of
158// void(const Packet&). The input side packet must have the tag "CALLBACK" in
159// the graph config.
160//
161// Example Usage:
162//
163// // Callback function.
164// void MyClass::MyFunction(const Packet& packet) {
165// count_ += packet.Get<int>();
166// }
167//
168// void MyClass::Run() {
169// CalculatorGraphConfig config;
170// LoadPartialConfigSomehow(&config);
171// std::string input_side_packet_name;
172// tool::AddCallbackCalculator("the_output_stream", &config,
173// &input_side_packet_name, true);
174// CalculatorGraph graph(config);
175// ABSL_CHECK_OK(graph.Run(
176// {{input_side_packet_name,
177// MakePacket<std::function<void(const Packet&)>>(
178// std::bind(&MyClass::MyFunction, this, std::placeholders::_1))}}
179// ));
180// }
181class CallbackCalculator : public CalculatorBase {
182 public:
183 CallbackCalculator() {}
184 CallbackCalculator(const CallbackCalculator&) = delete;
185 CallbackCalculator& operator=(const CallbackCalculator&) = delete;
186
187 ~CallbackCalculator() override {}
188
189 static absl::Status GetContract(CalculatorContract* cc);
190
191 absl::Status Open(CalculatorContext* cc) override;
192 absl::Status Process(CalculatorContext* cc) override;
193
194 private:
195 std::function<void(const Packet&)> callback_;
196 std::function<void(const std::vector<Packet>&)> vector_callback_;
197};

213 private:
214 std::function<void(const Packet&, const Packet&)> callback_;
215 // The header packet read from the stream.
216 // Header packet is only going to be read once, either during the Open() for
217 // the current implementation, or in the Process() call when the header stream
218 // has the packet.
219 Packet header_packet_;
220};
