28// A thread pool consists of a set of threads that sit around waiting
29// for callbacks to appear on a queue. When that happens, one of the
30// threads pulls a callback off the queue and runs it.
31//
32// The thread pool is shut down when the pool is destroyed.
33//
34// Sample usage:
35//
36// {
37// ThreadPool pool("testpool", num_workers);
38// pool.StartWorkers();
39// for (int i = 0; i < N; ++i) {
40// pool.Schedule([i]() { DoWork(i); });
41// }
42// }
43//
44class ThreadPool {
45 public:
46 // Create a thread pool that provides a concurrency of "num_threads"
47 // threads. I.e., if "num_threads" items are added, they are all
48 // guaranteed to run concurrently without excessive delay.
49 // It has an effectively infinite maximum queue length.
50 // If num_threads is 1, the callbacks are run in FIFO order.
51 explicit ThreadPool(int num_threads);
52 ThreadPool(const ThreadPool&) = delete;
53 ThreadPool& operator=(const ThreadPool&) = delete;
54
55 // Like the ThreadPool(int num_threads) constructor, except that
56 // it also associates "name_prefix" with each of the threads
57 // in the thread pool.
58 ThreadPool(const std::string& name_prefix, int num_threads);
59
60 // Create a thread pool that creates and can use up to "num_threads"
61 // threads. Any standard thread options, such as stack size, should
62 // be passed via "thread_options". "name_prefix" specifies the
63 // thread name prefix.
64 ThreadPool(const ThreadOptions& thread_options,
65 const std::string& name_prefix, int num_threads);
66
67 // Waits for closures (if any) to complete. May be called without
68 // having called StartWorkers().
69 ~ThreadPool();
70
71 // REQUIRES: StartWorkers has not been called
72 // Actually start the worker threads.
73 void StartWorkers();
74
75 // REQUIRES: StartWorkers has been called
76 // Add specified callback to queue of pending callbacks. Eventually a
77 // thread will pull this callback off the queue and execute it.
78 void Schedule(std::function<void()> callback);
79
80 // Provided for debugging and testing only.
81 int num_threads() const;
82
83 // Standard thread options. Use this accessor to get them.
84 const ThreadOptions& thread_options() const;
85
86 private:
87 class WorkerThread;
88 void RunWorker();
89
90 std::string name_prefix_;
91 std::vector<WorkerThread*> threads_;
92 int num_threads_;
93
94 absl::Mutex mutex_;
95 absl::CondVar condition_;
96 bool stopped_ ABSL_GUARDED_BY(mutex_) = false;
97 std::deque<std::function<void()>> tasks_ ABSL_GUARDED_BY(mutex_);
98
99 ThreadOptions thread_options_;
100};
