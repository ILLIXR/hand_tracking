44// TODO: rename to GpuService or GpuManager or something.
45class GpuResources {
46 public:
47 using StatusOrGpuResources = absl::StatusOr<std::shared_ptr<GpuResources>>;
48
49 static StatusOrGpuResources Create();
50 static StatusOrGpuResources Create(PlatformGlContext external_context);
51
52 // The destructor must be defined in the implementation file so that on iOS
53 // the correct ARC release calls are generated.
54 ~GpuResources();
55
56 explicit GpuResources(PlatformGlContext external_context);
57
58 // Shared GL context for calculators.
59 // TODO: require passing a context or node identifier.
60 const std::shared_ptr<GlContext>& gl_context() { return gl_context(nullptr); }
61
62 const std::shared_ptr<GlContext>& gl_context(CalculatorContext* cc);
63
64 // Shared buffer pool.
65 GpuBufferMultiPool& gpu_buffer_pool() { return gpu_buffer_pool_; }
66
67#ifdef __APPLE__
68 MetalSharedResources& metal_shared() { return *metal_shared_; }
69#endif // defined(__APPLE__)§
70
71 absl::Status PrepareGpuNode(CalculatorNode* node);
72
73 // If the node requires custom GPU executors in the current configuration,
74 // returns the executor's names and the executors themselves.
75 const std::map<std::string, std::shared_ptr<Executor>>& GetGpuExecutors() {
76 return named_executors_;
77 }
78
79 private:
80 GpuResources() = delete;
81 explicit GpuResources(std::shared_ptr<GlContext> gl_context);
82
83 GlContext::StatusOrGlContext GetOrCreateGlContext(const std::string& key);
84 const std::string& ContextKey(const std::string& canonical_node_name);
85
86 std::map<std::string, std::string> node_key_;
87
88 using GlContextMapType = std::map<std::string, std::shared_ptr<GlContext>>;
89 std::unique_ptr<GlContextMapType, void (*)(GlContextMapType*)>
90 gl_key_context_;
91
92#ifdef MEDIAPIPE_GPU_BUFFER_USE_CV_PIXEL_BUFFER
93 std::shared_ptr<CvTextureCacheManager> texture_caches_;
94#endif // MEDIAPIPE_GPU_BUFFER_USE_CV_PIXEL_BUFFER
95
96 // The pool must be destructed before the gl_context, but after the
97 // ios_gpu_data, so the declaration order is important.
98 GpuBufferMultiPool gpu_buffer_pool_;
99
100#ifdef __APPLE__
101 std::unique_ptr<MetalSharedResources> metal_shared_;
102#endif // defined(__APPLE__)
103
104 std::map<std::string, std::shared_ptr<Executor>> named_executors_;
105};

107// Legacy struct to keep existing client code happy.
108// TODO: eliminate!
109struct GpuSharedData {
110 GpuSharedData();
111
112 explicit GpuSharedData(PlatformGlContext external_context)
113 : GpuSharedData(CreateGpuResourcesOrDie(external_context)) {}
114
115 explicit GpuSharedData(std::shared_ptr<GpuResources> gpu_resources)
116 : gpu_resources(gpu_resources),
117 gl_context(gpu_resources->gl_context()),
118 gpu_buffer_pool(gpu_resources->gpu_buffer_pool()) {}
119
120 std::shared_ptr<GpuResources> gpu_resources;
121
122 std::shared_ptr<GlContext> gl_context;
123
124 GpuBufferMultiPool& gpu_buffer_pool;
125
126 private:
127 static std::shared_ptr<GpuResources> CreateGpuResourcesOrDie(
128 PlatformGlContext external_context) {
129 auto status_or_resources = GpuResources::Create(external_context);
130 MEDIAPIPE_CHECK_OK(status_or_resources.status())
131 << ": could not create GpuResources";
132 return std::move(status_or_resources).value();
133 }
134};
