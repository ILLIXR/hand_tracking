35// Experimental: CalculatorBase will eventually replace Calculator as the
36// base class of leaf (non-subgraph) nodes in a CalculatorGraph.
37//
38// The base calculator class. A subclass must, at a minimum, provide the
39// implementation of GetContract(), Process(), and register the calculator
40// using REGISTER_CALCULATOR(MyClass).
41//
42// The framework calls four primary functions on a calculator.
43// On initialization of the graph, a static function is called.
44// GetContract()
45// Then, for each run of the graph on a set of input side packets, the
46// following sequence will occur.
47// Open()
48// Process() (repeatedly)
49// Close()
50//
51// The entire calculator is constructed and destroyed for each graph run
52// (set of input side packets, which could mean once per video, or once
53// per image). Any expensive operations and large objects should be
54// input side packets.
55//
56// The framework calls Open() to initialize the calculator.
57// If appropriate, Open() should call cc->SetOffset() or
58// cc->Outputs().Get(id)->SetNextTimestampBound() to allow the framework to
59// better optimize packet queueing.
60//
61// The framework calls Process() for every packet received on the input
62// streams. The framework guarantees that cc->InputTimestamp() will
63// increase with every call to Process(). An empty packet will be on the
64// input stream if there is no packet on a particular input stream (but
65// some other input stream has a packet).
66//
67// The framework calls Close() after all calls to Process().
68//
69// Calculators with no inputs are referred to as "sources" and are handled
70// slightly differently than non-sources (see the function comments for
71// Process() for more details).
72//
73// Calculators must be thread-compatible.
74// The framework does not call the non-const methods of a calculator from
75// multiple threads at the same time. However, the thread that calls the
76// methods of a calculator is not fixed. Therefore, calculators should not
77// use ThreadLocal objects.
78class CalculatorBase {
79 public:
80 CalculatorBase();
81 virtual ~CalculatorBase();
82
83 // The subclasses of CalculatorBase must implement GetContract.
84 // The calculator cannot be registered without it. Notice that although
85 // this function is static the registration macro provides access to
86 // each subclass' GetContract function.
87 //
88 // static absl::Status GetContract(CalculatorContract* cc);
89 //
90 // GetContract fills in the calculator's contract with the framework, such
91 // as its expectations of what packets it will receive. When this function
92 // is called, the numbers of inputs, outputs, and input side packets will
93 // have already been determined by the calculator graph. You can use
94 // indexes, tags, or tag:index to access input streams, output streams,
95 // or input side packets.
96 //
97 // Example (uses tags for inputs and indexes for outputs and input side
98 // packets):
99 // cc->Inputs().Tag("VIDEO").Set<ImageFrame>("Input Image Frames.");
100 // cc->Inputs().Tag("AUDIO").Set<Matrix>("Input Audio Frames.");
101 // cc->Outputs().Index(0).Set<Matrix>("Output FooBar feature.");
102 // cc->InputSidePackets().Index(0).Set<MyModel>(
103 // "Model used for FooBar feature extraction.");
104 //
105 // Example (same number and type of outputs as inputs):
106 // for (int i = 0; i < cc->Inputs().NumEntries(); ++i) {
107 // // SetAny() is used to specify that whatever the type of the
108 // // stream is, it's acceptable. This does not mean that any
109 // // packet is acceptable. Packets in the stream still have a
110 // // particular type. SetAny() has the same effect as explicitly
111 // // setting the type to be the stream's type.
112 // cc->Inputs().Index(i).SetAny(StrCat("Generic Input Stream ", i));
113 // // Set each output to accept the same specific type as the
114 // // corresponding input.
115 // cc->Outputs().Index(i).SetSameAs(
116 // &cc->Inputs().Index(i), StrCat("Generic Output Stream ", i));
117 // }
118
119 // Open is called before any Process() calls, on a freshly constructed
120 // calculator. Subclasses may override this method to perform necessary
121 // setup, and possibly output Packets and/or set output streams' headers.
122 // Must return absl::OkStatus() to indicate success. On failure any
123 // other status code can be returned. If failure is returned then the
124 // framework will call neither Process() nor Close() on the calculator (so any
125 // necessary cleanup should be done before returning failure or in the
126 // destructor).
127 virtual absl::Status Open(CalculatorContext* cc) {UNUSED(cc); return absl::OkStatus(); }
128
129 // Processes the incoming inputs. May call the methods on cc to access
130 // inputs and produce outputs.
131 //
132 // Process() called on a non-source node must return
133 // absl::OkStatus() to indicate that all went well, or any other
134 // status code to signal an error.
135 // For example:
136 // absl::UnknownError("Failure Message");
137 // Notice the convenience functions in util/task/canonical_errors.h .
138 // If a non-source Calculator returns tool::StatusStop(), then this
139 // signals the graph is being cancelled early. In this case, all
140 // source Calculators and graph input streams will be closed (and
141 // remaining Packets will propagate through the graph).
142 //
143 // A source node will continue to have Process() called on it as long
144 // as it returns absl::OkStatus(). To indicate that there is
145 // no more data to be generated return tool::StatusStop(). Any other
146 // status indicates an error has occurred.
147 virtual absl::Status Process(CalculatorContext* cc) = 0;
148
149 // Is called if Open() was called and succeeded. Is called either
150 // immediately after processing is complete or after a graph run has ended
151 // (if an error occurred in the graph). Must return absl::OkStatus()
152 // to indicate success. On failure any other status code can be returned.
153 // Packets may be output during a call to Close(). However, output packets
154 // are silently discarded if Close() is called after a graph run has ended.
155 //
156 // NOTE: Do not call cc->GraphStatus() in Close() if you need to check if the
157 // processing is complete. Please, see CalculatorContext::GraphStatus
158 // documentation for the suggested solution.
159 virtual absl::Status Close(CalculatorContext* cc) {UNUSED(cc); return absl::OkStatus(); }
160
161 // Returns a value according to which the framework selects
162 // the next source calculator to Process(); smaller value means
163 // Process() first. The default implementation returns the smallest
164 // NextTimestampBound value over all the output streams, but subclasses
165 // may override this. If a calculator is not a source, this method is
166 // not called.
167 // TODO: Does this method need to be virtual? No Calculator
168 // subclasses override the SourceProcessOrder method.
169 virtual Timestamp SourceProcessOrder(const CalculatorContext* cc) const;
170};

178// Gives access to the static functions within subclasses of CalculatorBase.
179// This adds functionality akin to virtual static functions.
180class CalculatorBaseFactory {
181 public:
182 virtual ~CalculatorBaseFactory() {}
183 virtual absl::Status GetContract(CalculatorContract* cc) = 0;
184 virtual std::unique_ptr<CalculatorBase> CreateCalculator(
185 CalculatorContext* calculator_context) = 0;
186 virtual std::string ContractMethodName() { return "GetContract"; }
187};
188
189// Functions for checking that the calculator has the required GetContract.
190template <class T>
191constexpr bool CalculatorHasGetContract(decltype(&T::GetContract) /*unused*/) {
192 typedef absl::Status (*GetContractType)(CalculatorContract* cc);
193 return std::is_same<decltype(&T::GetContract), GetContractType>::value;
194}
195template <class T>
196constexpr bool CalculatorHasGetContract(...) {
197 return false;
198}
199
200// Provides access to the static functions within a specific subclass
201// of CalculatorBase.
202template <class T, class Enable = void>
203class CalculatorBaseFactoryFor : public CalculatorBaseFactory {
204 static_assert(std::is_base_of<mediapipe::CalculatorBase, T>::value,
205 "Classes registered with REGISTER_CALCULATOR must be "
206 "subclasses of mediapipe::CalculatorBase.");
207};
208
209template <class T>
210class CalculatorBaseFactoryFor<
211 T,
212 typename std::enable_if<std::is_base_of<mediapipe::CalculatorBase, T>{} &&
213 !std::is_base_of<mediapipe::api2::Node, T>{}>::type>
214 : public CalculatorBaseFactory {
215 public:
216 static_assert(CalculatorHasGetContract<T>(nullptr),
217 "GetContract() must be defined with the correct signature in "
218 "every calculator.");
219
220 // Provides access to the static function GetContract within a specific
221 // subclass of CalculatorBase.
222 absl::Status GetContract(CalculatorContract* cc) final {
223 // CalculatorBaseSubclass must implement this function, since it is not
224 // implemented in the parent class.
225 return T::GetContract(cc);
226 }
227
228 std::unique_ptr<CalculatorBase> CreateCalculator(
229 CalculatorContext* calculator_context) final {
230 UNUSED(calculator_context);
231 return absl::make_unique<T>();
232 }
233};
