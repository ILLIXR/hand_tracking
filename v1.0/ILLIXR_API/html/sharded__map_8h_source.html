18#include <stddef.h>

20#include <unordered_map>

23#include "absl/synchronization/mutex.h"

25// A thread-safe unordered map with locking at the key level.
26template <typename Key, typename T, class Hash = std::hash<Key>>
27class ShardedMap {
28 public:
29 using Map = std::unordered_map<Key, T, Hash>;
30 using hasher = typename Map::hasher;
31 using value_type = typename Map::value_type;
32 template <typename ShardedMapPtr, class map_iterator>
33 class Iterator;
34 using iterator = Iterator<ShardedMap*, typename Map::iterator>;
35 using const_iterator =
36 Iterator<const ShardedMap*, typename Map::const_iterator>;
37
38 // Creates a ShardedMap to hold |size| elements in |num_shards| partitions.
39 ShardedMap(size_t capacity, size_t num_shards)
40 : maps_(num_shards, Map(capacity / num_shards)),
41 mutexes_(num_shards),
42 size_(0) {}
43
44 // Creates a ShardedMap to hold approximately |size| elements.
45 // Default capacity is 100, which avoids most lock contention.
46 explicit ShardedMap(size_t capacity = 100)
47 : ShardedMap(capacity, capacity / 10 + 1) {}
48
49 // Returns the iterator to the entry for a key.
50 inline iterator find(const Key& key) ABSL_NO_THREAD_SAFETY_ANALYSIS {
51 size_t shard = Index(key);
52 mutexes_[shard].Lock();
53 typename Map::iterator iter = maps_[shard].find(key);
54 if (iter == maps_[shard].end()) {
55 mutexes_[shard].Unlock();
56 return end();
57 }
58 return {shard, iter, this};
59 }
60
61 // Returns 1 if the container includes a certain key.
62 inline size_t count(const Key& key) const {
63 size_t shard = Index(key);
64 absl::MutexLock lock(&mutexes_[shard]);
65 return maps_[shard].count(key);
66 }
67
68 // Adds an entry to the map and returns the iterator to it.
69 inline std::pair<iterator, bool> insert(const value_type& val)
70 ABSL_NO_THREAD_SAFETY_ANALYSIS {
71 size_t shard = Index(val.first);
72 mutexes_[shard].Lock();
73 std::pair<typename Map::iterator, bool> p = maps_[shard].insert(val);
74 size_ += p.second ? 1 : 0;
75 return {std::move(iterator{shard, p.first, this}), p.second};
76 }
77
78 // Removes the entry for an iterator.
79 inline void erase(iterator& pos) {
80 if (pos != end()) {
81 auto next_iter = pos.iter_;
82 next_iter++;
83 maps_[pos.shard_].erase(pos.iter_);
84 pos.iter_ = next_iter;
85 pos.NextEntryShard();
86 --size_;
87 }
88 }
89
90 // The total count of entries.
91 inline size_t size() const { return size_; }
92
93 // Returns the iterator to the first element.
94 inline iterator begin() ABSL_NO_THREAD_SAFETY_ANALYSIS {
95 mutexes_[0].Lock();
96 iterator result{0, maps_[0].begin(), this};
97 result.NextEntryShard();
98 return result;
99 }
100
101 // Returns the iterator after the last element.
102 // The end() iterator doesn't belong to any shard.
103 inline iterator end() {
104 return iterator{maps_.size() - 1, maps_.back().end(), this};
105 }
106
107 inline const_iterator begin() const {
108 mutexes_[0].Lock();
109 const_iterator result{0, maps_[0].begin(), this};
110 result.NextEntryShard();
111 return result;
112 }
113 inline const_iterator end() const {
114 return const_iterator{maps_.size() - 1, maps_.back().end(), this};
115 }
116
117 // The iterator across map entries.
118 // The iterator keeps its shard locked until it is destroyed.
119 template <typename ShardedMapPtr, class map_iterator>
120 class Iterator {
121 public:
122 Iterator(Iterator&& other)
123 : shard_(other.shard_), iter_(other.iter_), map_(other.map_) {
124 other.map_ = nullptr;
125 }
126 ~Iterator() { Clear(); }
127 Iterator& operator=(Iterator&& other) {
128 Clear();
129 shard_ = other.shard_, iter_ = other.iter_, map_ = other.map_;
130 other.map_ = nullptr;
131 return *this;
132 }
133 inline bool operator==(const Iterator& other) const {
134 return shard_ == other.shard_ && iter_ == other.iter_;
135 }
136 inline bool operator!=(const Iterator& other) const {
137 return !operator==(other);
138 }
139 inline typename std::iterator_traits<map_iterator>::reference operator*()
140 const {
141 return *iter_;
142 }
143 inline typename std::iterator_traits<map_iterator>::pointer operator->()
144 const {
145 return &(operator*());
146 }
147 inline void operator++() {
148 iter_++;
149 NextEntryShard();
150 }
151
152 private:
153 Iterator(size_t shard, map_iterator iter, ShardedMapPtr map)
154 : shard_(shard), iter_(iter), map_(map) {}
155 // Releases all resources.
156 inline void Clear() ABSL_NO_THREAD_SAFETY_ANALYSIS {
157 if (!map_) return;
158 bool is_end = (shard_ == map_->maps_.size() - 1 &&
159 iter_ == map_->maps_[shard_].end());
160 if (!is_end) {
161 map_->mutexes_[shard_].Unlock();
162 }
163 map_ = nullptr;
164 }
165 // Moves to the shard of the next entry.
166 void NextEntryShard() ABSL_NO_THREAD_SAFETY_ANALYSIS {
167 size_t last = map_->maps_.size() - 1;
168 while (iter_ == map_->maps_[shard_].end() && shard_ < last) {
169 map_->mutexes_[shard_].Unlock();
170 shard_++;
171 map_->mutexes_[shard_].Lock();
172 iter_ = map_->maps_[shard_].begin();
173 }
174 if (iter_ == map_->maps_.back().end()) {
175 map_->mutexes_[shard_].Unlock();
176 }
177 }
178 size_t shard_;
179 map_iterator iter_;
180 ShardedMapPtr map_;
181 friend ShardedMap;
182 };
183
184 private:
185 // Returns the shard index for a key.
186 inline size_t Index(const Key& key) const {
187 return hasher{}(key) % maps_.size();
188 }
189
190 // One unordered map for each key shard.
191 std::vector<Map> maps_;
192
193 // One mutex for each key shard.
194 mutable std::vector<absl::Mutex> mutexes_;
195
196 // The total count of entries.
197 std::atomic<int> size_;
198};
