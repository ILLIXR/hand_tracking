39class ProfilingContext;
40// Holds data that the Calculator needs access to. This data is not
41// stored in Calculator directly since Calculator will be destroyed after
42// every CalculatorGraph::Run() . It is not stored in CalculatorNode
43// because Calculator should not depend on CalculatorNode. All
44// information conveyed in this class is flowing from the CalculatorNode
45// to the Calculator.
46class CalculatorState {
47 public:
48 CalculatorState(const std::string& node_name, int node_id,
49 const std::string& calculator_type,
50 const CalculatorGraphConfig::Node& node_config,
51 std::shared_ptr<ProfilingContext> profiling_context);
52 CalculatorState(const CalculatorState&) = delete;
53 CalculatorState& operator=(const CalculatorState&) = delete;
54 ~CalculatorState();
55
56 // Called before every call to Calculator::Open() (during the PrepareForRun
57 // phase).
58 void ResetBetweenRuns();
59
60 const std::string& CalculatorType() const { return calculator_type_; }
61 const CalculatorOptions& Options() const { return node_config_.options(); }
62 // Returns the options given to this calculator. Template argument T must
63 // be the type of the protobuf extension message or the protobuf::Any
64 // message containing the options.
65 template <class T>
66 const T& Options() const {
67 return options_.Get<T>();
68 }
69 template <class T>
70 bool HasOptions() const {
71 return options_.Has<T>();
72 }
73 const std::string& NodeName() const { return node_name_; }
74 const int& NodeId() const { return node_id_; }
75
77 // Interface for Calculator.
79 const PacketSet& InputSidePackets() const { return *input_side_packets_; }
80 OutputSidePacketSet& OutputSidePackets() { return *output_side_packets_; }
81
82 // Returns a counter using the graph's counter factory. The counter's
83 // name is the passed-in name, prefixed by the calculator NodeName.
84 Counter* GetCounter(const std::string& name);
85
86 // Returns a counter set, which can be passed to other classes, to generate
87 // counters. NOTE: This differs from GetCounter, in that the counters
88 // created by this counter set do not have the NodeName prefix.
89 CounterFactory* GetCounterFactory();
90
91 std::shared_ptr<ProfilingContext> GetSharedProfilingContext() const {
92 return profiling_context_;
93 }
94
96 // Interface for CalculatorNode.
98 // Sets the input side packets.
99 void SetInputSidePackets(const PacketSet* input_side_packets);
100 // Sets the output side packets.
101 void SetOutputSidePackets(OutputSidePacketSet* output_side_packets);
102 // Sets the counter factory.
103 void SetCounterFactory(CounterFactory* counter_factory) {
104 counter_factory_ = counter_factory;
105 }
106
107 absl::Status SetServicePacket(const GraphServiceBase& service,
108 Packet packet) {
109 return graph_service_manager_.SetServicePacket(service, packet);
110 }
111
112 template <typename T>
113 std::shared_ptr<T> GetServiceObject(const GraphService<T>& service) {
114 return graph_service_manager_.GetServiceObject(service);
115 }
116
117 private:
119 // Persistent variables that are not cleared by ResetBetweenRuns().
121 // The name associated with this calculator's node.
122 const std::string node_name_;
123 // The ID associated with this calculator's node.
124 const int node_id_;
125 // The registered type name of the Calculator.
126 const std::string calculator_type_;
127 // The Node protobuf containing the options for the calculator.
128 const CalculatorGraphConfig::Node node_config_;
129 // The unpacked protobuf options for the calculator.
130 tool::OptionsMap options_;
131 // The graph tracing and profiling interface.
132 std::shared_ptr<ProfilingContext> profiling_context_;
133
134 GraphServiceManager graph_service_manager_;
135
137 // Variables which ARE cleared by ResetBetweenRuns().
139 // The set of input side packets set by CalculatorNode::PrepareForRun().
140 // ResetBetweenRuns() clears this PacketSet pointer.
141 const PacketSet* input_side_packets_;
142 // The OutputSidePacketSet object is owned by the CalculatorNode.
143 // CalculatorState obtains its pointer in CalculatorNode::PrepareForRun.
144 OutputSidePacketSet* output_side_packets_;
145
146 CounterFactory* counter_factory_;
147};
