43// Encapsulates the type and description of an input stream, output
44// stream, or input side packet. The type to expect is set with Set<type>()
45// and a packet is validated with Validate(). The PacketType can be
46// set to indicate the same type as some other PacketType.
47class PacketType {
48 public:
49 // Creates an uninitialized PacketType.
50 PacketType() = default;
51
52 // PacketType can be passed by value.
53 PacketType(const PacketType&) = default;
54 PacketType& operator=(const PacketType&) = default;
55
56 // False for a PacketType that has not had any Set*() function called.
57 bool IsInitialized() const;
58
59 // The following "Set*" functions initialize the PacketType. They can
60 // be called any number of times; the PacketType is initialized to
61 // whatever the last call is.
62
63 // Sets the packet type to accept the provided type.
64 template <typename T>
65 PacketType& Set();
66 // Sets the packet type to accept packets of any type. If an input or
67 // output stream is set to this type then the framework tries to infer
68 // the stream type based on the packet types of other Calculators.
69 // Specifically, using SetAny() still means that the stream has a type
70 // but this particular calculator just doesn't care what it is.
71 PacketType& SetAny();
72 // Sets the packet type to accept any of the provided types.
73 template <typename... T>
74 PacketType& SetOneOf();
75 // Sets the packet type to not accept any packets.
76 PacketType& SetNone();
77 // Sets the PacketType to be the same as type. This actually stores
78 // the pointer to the other PacketType but does not acquire ownership.
79 // "type" must outlive this object.
80 PacketType& SetSameAs(const PacketType* type);
81 // Marks this port as optional.
82 PacketType& Optional();
83
84 // Returns a pointer to the root PacketType of the "same as" equivalence
85 // class.
86 const PacketType* GetSameAs() const;
87 PacketType* GetSameAs();
88 // Returns true if this PacketType allows anything.
89 bool IsAny() const;
90 // Returns true if this PacketType allows nothing.
91 bool IsNone() const;
92 // Returns true if this PacketType allows a set of types.
93 bool IsOneOf() const;
94 // Returns true if this PacketType allows one specific type.
95 bool IsExactType() const;
96 // Returns true if this port has been marked as optional.
97 bool IsOptional() const { return optional_; }
98
99 // Returns true iff this and other are consistent, meaning they do
100 // not expect different types. IsAny() is consistent with anything.
101 // IsNone() is only consistent with IsNone() and IsAny().
102 // Note: this is definied as a symmetric relationship, but within the
103 // framework, it is consistently invoked as:
104 // input_port_type.IsConsistentWith(connected_output_port_type)
105 // TODO: consider making this explicitly directional, and
106 // sharing some logic with the packet validation check.
107 bool IsConsistentWith(const PacketType& other) const;
108
109 // Returns OK if the packet contains an object of the appropriate type.
110 absl::Status Validate(const Packet& packet) const;
111
112 // Returns a pointer to the Registered type name, or nullptr if the type
113 // is not registered. Do not use this for validation, use Validate()
114 // instead.
115 const std::string* RegisteredTypeName() const;
116 // Returns the type name. Do not use this for validation, use
117 // Validate() instead.
118 std::string DebugTypeName() const;
119
120 private:
121 struct SameAs {
122 // This PacketType is the same as other.
123 // We don't do union-find optimizations in order to avoid a mutex.
124 const PacketType* other;
125 };
126 using TypeIdSpan = absl::Span<const TypeId>;
127 struct MultiType {
128 TypeIdSpan types;
129 // TODO: refactor RegisteredTypeName, remove.
130 const std::string* registered_type_name;
131 };
132 struct SpecialType;
133 using TypeSpec =
134 absl::variant<absl::monostate, TypeId, MultiType, SameAs, SpecialType>;
135 typedef absl::Status (*AcceptsTypeFn)(const TypeSpec& type);
136 struct SpecialType {
137 std::string name_;
138 AcceptsTypeFn accept_fn_;
139 };
140
141 static absl::Status AcceptAny(const TypeSpec& type);
142 static absl::Status AcceptNone(const TypeSpec& type);
143
144 const PacketType* SameAsPtr() const;
145 static TypeIdSpan GetTypeSpan(const TypeSpec& type_spec);
146 static std::string TypeNameForOneOf(TypeIdSpan types);
147
148 TypeSpec type_spec_;
149
150 // Whether the corresponding port is optional.
151 bool optional_ = false;
152};

154// An error handler class which allows a PacketTypeSet to return valid
155// results while deferring errors.
156//
157// This class is thread compatible.
158class PacketTypeSetErrorHandler {
159 public:
160 // Returns a usable PacketType. A different PacketType object is
161 // returned for each different invalid location and the same object
162 // is returned for multiple accesses to the same invalid location.
163 PacketType& GetFallback(const absl::string_view tag, int index) {
164 if (!missing_) {
165 missing_ = absl::make_unique<Missing>();
166 }
167 ABSL_CHECK(!missing_->initialized_errors);
168 std::string key = absl::StrCat(tag, ":", index);
169 return missing_->entries[key];
170 }
171
172 // In the const setting produce a FATAL error.
173 const PacketType& GetFallback(const absl::string_view tag, int index) const {
174 ABSL_LOG(FATAL) << "Failed to get tag \"" << tag << "\" index " << index
175 << ". Unable to defer error due to const specifier.";
176 std::abort();
177 }
178
179 // Returns true if a (deferred) error has been recorded by the
180 // PacketTypeSet.
181 bool HasError() const { return missing_ != nullptr; }
182
183 // Get the error messages that have been deferred.
184 // This function can only be called if HasError() is true.
185 const std::vector<std::string>& ErrorMessages() const {
186 ABSL_CHECK(missing_) << "ErrorMessages() can only be called if errors have "
187 "occurred. Call HasError() before calling this "
188 "function.";
189 if (!missing_->initialized_errors) {
190 for (const auto& entry : missing_->entries) {
191 // Optional entries that were missing are not considered errors.
192 if (!entry.second.IsOptional()) {
193 // Split them to keep the error string unchanged.
194 std::pair<std::string, std::string> tag_idx =
195 absl::StrSplit(entry.first, ':');
196 missing_->errors.push_back(absl::StrCat("Failed to get tag \"",
197 tag_idx.first, "\" index ",
198 tag_idx.second));
199 }
200 }
201 missing_->initialized_errors = true;
202 }
203 return missing_->errors;
204 }
205
206 private:
207 struct Missing {
208 // Mapping from TAG:index to PacketType objects, one for each invalid
209 // location that has been accessed.
210 std::map<std::string, PacketType> entries;
211 // The list of errors is only computed at the end.
212 std::vector<std::string> errors;
213 bool initialized_errors = false;
214 };
215
216 // Initialize lazily to save space in the common no-error case.
217 std::unique_ptr<Missing> missing_;
218};

220// A collection of PacketTypes. The types are either retrieved by index
221// or by tag. The calculator must set a type for every input stream and
222// input side packet that it accepts and every output stream it produces.
223// Every (non-const) function in this class always returns valid values
224// that can be used directly without error checking. If the types don't
225// match what the user provided then an error will be triggered later
226// (but the program will not crash).
227//
228// For example, a calculator can just call
229// inputs->Tag("VIDEO").Set<ImageFrame>("Annotated Video Frames.");
230// Without checking that "VIDEO" is a valid tag.
231//
232// Similarly if the following is specified:
233// inputs->Index(0).Set<int>("Some Integer.");
234// inputs->Index(1).Set<std::string>("Some String.");
235// inputs->Index(2).Set<float>("Some Float.");
236// then it is not necessary to check that NumEntries() == 3. An error
237// is triggered if there aren't exactly 3 inputs or they don't have the
238// proper types.
239//
240// For a const PacketTypeSet, invalid access is a fatal error.
241//
242// This class is thread compatible.
243using PacketTypeSet =
244 internal::Collection<PacketType, internal::CollectionStorage::kStoreValue,
245 PacketTypeSetErrorHandler>;
246
247// Returns OK if the packets in the PacketSet are of the appropriate type.
248// packet_type_set must be valid before this is called (but packet_set
249// may be in any state).
250absl::Status ValidatePacketSet(const PacketTypeSet& packet_type_set,
251 const PacketSet& packet_set);
252
253// Validates that the PacketTypeSet was initialized properly.
254// An error is returned if
255// 1) Tag() or Index() is called with an invalid argument (however,
256// a valid PacketType is still returned by the function).
257// 2) Any PacketType is not initialized.
258absl::Status ValidatePacketTypeSet(const PacketTypeSet& packet_type_set);

260// Templated function definitions.
261
262template <typename T>
263PacketType& PacketType::Set() {
264 type_spec_ = kTypeId<T>;
265 return *this;
266}
267
268template <typename... T>
269PacketType& PacketType::SetOneOf() {
270 static const NoDestructor<std::vector<TypeId>> types{{kTypeId<T>...}};
271 static const NoDestructor<std::string> name{TypeNameForOneOf(*types)};
272 type_spec_ = MultiType{*types, &*name};
273 return *this;
274}
