43// Abstract base class for output stream handlers.
44class OutputStreamHandler {
45 public:
46 // Handy typedef for a map from the name of an output stream to the set of
47 // ids of upstream sources that affect it.
48 typedef std::unordered_map<std::string, std::unordered_set<int>>
49 OutputStreamToSourcesMap;
50 typedef internal::Collection<OutputStreamManager*> OutputStreamManagerSet;
51
52 // The constructor of the OutputStreamHandler takes four arguments.
53 // The tag_map argument holds the information needed for tag/index retrieval
54 // for the output streams; the calculator_context_manager for managing the
55 // calculator context objects; the options argument varies for different
56 // output stream handler subclasses; the calculator_run_in_parallel argument
57 // indicates the execution mode (sequential executions vs parallel executions)
58 // of the calcualtor;
59 OutputStreamHandler(std::shared_ptr<tool::TagMap> tag_map,
60 CalculatorContextManager* calculator_context_manager,
61 const MediaPipeOptions& options,
62 bool calculator_run_in_parallel)
63 : output_stream_managers_(std::move(tag_map)),
64 calculator_context_manager_(calculator_context_manager),
65 options_(options),
66 calculator_run_in_parallel_(calculator_run_in_parallel) {
67 ABSL_CHECK(calculator_context_manager_);
68 }
69
70 virtual ~OutputStreamHandler() = default;
71
72 const MediaPipeOptions& Options() const { return options_; }
73
74 // Initializes the OutputStreamManagerSet object.
75 // flat_output_stream_managers is expected to point to a contiguous
76 // flat array with OutputStreamManagers corresponding to the id's in
77 // OutputStreamHandler::output_stream_managers_ (meaning it should
78 // point to somewhere in the middle of the main flat array of all
79 // output stream managers).
80 absl::Status InitializeOutputStreamManagers(
81 OutputStreamManager* flat_output_stream_managers);
82
83 // Sets up output shards by connecting to the managers.
84 absl::Status SetupOutputShards(OutputStreamShardSet* output_shards);
85
86 int NumOutputStreams() const { return output_stream_managers_.NumEntries(); }
87
88 // Returns the tag map of the output streams.
89 const std::shared_ptr<tool::TagMap>& OutputTagMap() const {
90 return output_stream_managers_.TagMap();
91 }
92
93 // Calls OutputStreamManager::PrepareForRun(error_callback) per stream, and
94 // resets data memebers.
95 void PrepareForRun(const std::function<void(absl::Status)>& error_callback)
96 ABSL_LOCKS_EXCLUDED(timestamp_mutex_);
97
98 // Marks the output streams as started and propagates any changes made in
99 // Calculator::Open().
100 void Open(OutputStreamShardSet* output_shards);
101
102 // Prepares the OutputStreamShardSet before a call to Calculator's Open(),
103 // Process(), or Close().
104 void PrepareOutputs(Timestamp input_timestamp,
105 OutputStreamShardSet* output_shards);
106
107 // Propagates timestamp directly if there is no ongoing parallel invocation.
108 // Otherwise, updates task_timestamp_bound_.
109 void UpdateTaskTimestampBound(Timestamp timestamp)
110 ABSL_LOCKS_EXCLUDED(timestamp_mutex_);
111
112 // Invoked after a call to Calculator::Process() function.
113 void PostProcess(Timestamp input_timestamp)
114 ABSL_LOCKS_EXCLUDED(timestamp_mutex_);
115
116 // Propagates the output shards and closes all managed output streams.
117 void Close(OutputStreamShardSet* output_shards);
118
119 // Returns the name of the first stream in the output stream manager
120 // collection for debugging purpose.
121 std::string FirstStreamName() const;
122
123 const OutputStreamManagerSet& OutputStreams() {
124 return output_stream_managers_;
125 }
126
127 protected:
128 // Checks if the given input bound should be propagated or not. If any output
129 // streams with OffsetEnabled() need to have the timestamp bounds updated,
130 // then propagates the timestamp bounds of all output streams with
131 // OffsetEnabled() by adding their offsets to the given input bound.
132 void TryPropagateTimestampBound(Timestamp input_bound);
133
134 // Computes the output timestamp bound and propagates output packets to
135 // downstream nodes.
136 void PropagateOutputPackets(Timestamp input_timestamp,
137 OutputStreamShardSet* output_shards);
138
139 // The packets and timestamp propagation logic for parallel execution.
140 virtual void PropagationLoop()
141 ABSL_EXCLUSIVE_LOCKS_REQUIRED(timestamp_mutex_) = 0;
142
143 // Collection of all OutputStreamManager objects.
144 OutputStreamManagerSet output_stream_managers_;
145 // A pointer to the calculator context manager of the calculator node.
146 CalculatorContextManager* const calculator_context_manager_;
147 MediaPipeOptions options_;
148 const bool calculator_run_in_parallel_;
149
150 absl::Mutex timestamp_mutex_;
151 // A set of the completed input timestamps in ascending order.
152 std::set<Timestamp> completed_input_timestamps_
153 ABSL_GUARDED_BY(timestamp_mutex_);
154 // The current minimum timestamp for which a new packet could possibly arrive.
155 // TODO: Rename the variable to be more descriptive.
156 Timestamp task_timestamp_bound_ ABSL_GUARDED_BY(timestamp_mutex_);
157
158 // PropagationState indicates the current state of the propagation process.
159 // There are eight possible transitions:
160 // (a) From kIdle to kPropagatingPackets.
161 // Any thread that makes this transition becomes the propagation thread, and
162 // will be responsible for propagating all available packets and the timestamp
163 // bound.
164 // (b) From kIdle to kPropagatingBound.
165 // Any thread that makes this transition becomes the propagation thread, and
166 // will be responsible for propagating all available packets and timestamp
167 // bound.
168 // (c) From kPropagatingPackets to kIdle.
169 // Made by the propagation thread after all available packets have been
170 // propagated. If timestamp bound propagation isn't necessary, the propagation
171 // process is considered to be completed.
172 // (d) From kPropagatingPackets to kPropagatingBound.
173 // Made by the propagation thread to indicate that packets propagation is
174 // completed, and it is going to do timestamp propagation without holding the
175 // timestamp_mutex_.
176 // (e) From kPropagatingBound to kPropagationPending.
177 // Any thread, except the propagation thread, can make this transition.
178 // kPropagationPending indicates that some recent changes require the
179 // propagation thread to recheck if any new packets or new timestamp bound can
180 // be propagated.
181 // (f) From kPropagationPending to kPropagatingPackets.
182 // Made by the propagation thread to mark that there are some new packets that
183 // need to be propagated.
184 // (g) From kPropagatingBound to kIdle.
185 // Made by the propagation thread when there is no more propagation work to
186 // be done.
187 // (h) From kPropagationPending to kIdle.
188 // Made by the propagation thread when the newly arrived packets and timestamp
189 // bound are still not ready to be propagated.
190 enum PropagationState {
191 kIdle = 0, //
192 kPropagatingPackets = 1, //
193 kPropagatingBound = 2, //
194 kPropagationPending = 3
195 };
196 PropagationState propagation_state_ ABSL_GUARDED_BY(timestamp_mutex_) = kIdle;
197};
