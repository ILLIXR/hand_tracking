36template <typename T>
37using IsCopyable = std::enable_if_t<std::is_copy_constructible<T>::value, bool>;
38
39template <typename T>
40using IsNotCopyable =
41 std::enable_if_t<!std::is_copy_constructible<T>::value, bool>;
42
43template <typename T>
44using IsMovable = std::enable_if_t<std::is_move_constructible<T>::value, bool>;

48 std::enable_if_t<!std::is_move_constructible<T>::value, bool>;

50// Splits an input packet with std::vector<T> into multiple std::vector<T>
51// output packets using the [begin, end) ranges specified in
52// SplitVectorCalculatorOptions. If the option "element_only" is set to true,
53// all ranges should be of size 1 and all outputs will be elements of type T. If
54// "element_only" is false, ranges can be non-zero in size and all outputs will
55// be of type std::vector<T>. If the option "combine_outputs" is set to true,
56// only one output stream can be specified and all ranges of elements will be
57// combined into one vector.
58// To use this class for a particular type T, register a calculator using
59// SplitVectorCalculator<T>.
60template <typename T, bool move_elements>
61class SplitVectorCalculator : public CalculatorBase {
62 public:
63 static absl::Status GetContract(CalculatorContract* cc) {
64 RET_CHECK(cc->Inputs().NumEntries() == 1);
65 RET_CHECK(cc->Outputs().NumEntries() != 0);
66
67 cc->Inputs().Index(0).Set<std::vector<T>>();
68
69 const auto& options =
70 cc->Options<::mediapipe::SplitVectorCalculatorOptions>();
71
72 if (!std::is_copy_constructible<T>::value || move_elements) {
73 // Ranges of elements shouldn't overlap when the vector contains
74 // non-copyable elements.
75 RET_CHECK_OK(checkRangesDontOverlap(options));
76 }
77
78 if (options.combine_outputs()) {
79 RET_CHECK_EQ(cc->Outputs().NumEntries(), 1);
80 cc->Outputs().Index(0).Set<std::vector<T>>();
81 RET_CHECK_OK(checkRangesDontOverlap(options));
82 } else {
83 if (cc->Outputs().NumEntries() != options.ranges_size()) {
84 return absl::InvalidArgumentError(
85 "The number of output streams should match the number of ranges "
86 "specified in the CalculatorOptions.");
87 }
88
89 // Set the output types for each output stream.
90 for (int i = 0; i < cc->Outputs().NumEntries(); ++i) {
91 if (options.ranges(i).begin() < 0 || options.ranges(i).end() < 0 ||
92 options.ranges(i).begin() >= options.ranges(i).end()) {
93 return absl::InvalidArgumentError(
94 "Indices should be non-negative and begin index should be less "
95 "than the end index.");
96 }
97 if (options.element_only()) {
98 if (options.ranges(i).end() - options.ranges(i).begin() != 1) {
99 return absl::InvalidArgumentError(
100 "Since element_only is true, all ranges should be of size 1.");
101 }
102 cc->Outputs().Index(i).Set<T>();
103 } else {
104 cc->Outputs().Index(i).Set<std::vector<T>>();
105 }
106 }
107 }
108
109 return absl::OkStatus();
110 }
111
112 absl::Status Open(CalculatorContext* cc) override {
113 cc->SetOffset(TimestampDiff(0));
114
115 const auto& options =
116 cc->Options<::mediapipe::SplitVectorCalculatorOptions>();
117
118 element_only_ = options.element_only();
119 combine_outputs_ = options.combine_outputs();
120
121 for (const auto& range : options.ranges()) {
122 ranges_.push_back({range.begin(), range.end()});
123 max_range_end_ = std::max(max_range_end_, range.end());
124 total_elements_ += range.end() - range.begin();
125 }
126
127 return absl::OkStatus();
128 }
129
130 absl::Status Process(CalculatorContext* cc) override {
131 if (cc->Inputs().Index(0).IsEmpty()) return absl::OkStatus();
132
133 if (move_elements) {
134 return ProcessMovableElements<T>(cc);
135 } else {
136 return ProcessCopyableElements<T>(cc);
137 }
138 }
139
140 template <typename U, IsCopyable<U> = true>
141 absl::Status ProcessCopyableElements(CalculatorContext* cc) {
142 // static_assert(std::is_copy_constructible<U>::value,
143 // "Cannot copy non-copyable elements");
144 const auto& input = cc->Inputs().Index(0).Get<std::vector<U>>();
145 RET_CHECK_GE((int)input.size(), max_range_end_);
146 if (combine_outputs_) {
147 auto output = absl::make_unique<std::vector<U>>();
148 output->reserve(total_elements_);
149 for (int i = 0; i < (int)ranges_.size(); ++i) {
150 auto elements = absl::make_unique<std::vector<U>>(
151 input.begin() + ranges_[i].first,
152 input.begin() + ranges_[i].second);
153 output->insert(output->end(), elements->begin(), elements->end());
154 }
155 cc->Outputs().Index(0).Add(output.release(), cc->InputTimestamp());
156 } else {
157 if (element_only_) {
158 for (int i = 0; i < (int)ranges_.size(); ++i) {
159 cc->Outputs().Index(i).AddPacket(
160 MakePacket<U>(input[ranges_[i].first]).At(cc->InputTimestamp()));
161 }
162 } else {
163 for (int i = 0; i < (int)ranges_.size(); ++i) {
164 auto output = absl::make_unique<std::vector<T>>(
165 input.begin() + ranges_[i].first,
166 input.begin() + ranges_[i].second);
167 cc->Outputs().Index(i).Add(output.release(), cc->InputTimestamp());
168 }
169 }
170 }
171
172 return absl::OkStatus();
173 }
174
175 template <typename U, IsNotCopyable<U> = true>
176 absl::Status ProcessCopyableElements(CalculatorContext* cc) {
177 UNUSED(cc);
178 return absl::InternalError("Cannot copy non-copyable elements.");
179 }
180
181 template <typename U, IsMovable<U> = true>
182 absl::Status ProcessMovableElements(CalculatorContext* cc) {
183 absl::StatusOr<std::unique_ptr<std::vector<U>>> input_status =
184 cc->Inputs().Index(0).Value().Consume<std::vector<U>>();
185 if (!input_status.ok()) return input_status.status();
186 std::unique_ptr<std::vector<U>> input_vector =
187 std::move(input_status).value();
188 RET_CHECK_GE((int)input_vector->size(), max_range_end_);
189
190 if (combine_outputs_) {
191 auto output = absl::make_unique<std::vector<U>>();
192 output->reserve(total_elements_);
193 for (int i = 0; i < (int)ranges_.size(); ++i) {
194 output->insert(
195 output->end(),
196 std::make_move_iterator(input_vector->begin() + ranges_[i].first),
197 std::make_move_iterator(input_vector->begin() + ranges_[i].second));
198 }
199 cc->Outputs().Index(0).Add(output.release(), cc->InputTimestamp());
200 } else {
201 if (element_only_) {
202 for (int i = 0; i < (int)ranges_.size(); ++i) {
203 cc->Outputs().Index(i).AddPacket(
204 MakePacket<U>(std::move(input_vector->at(ranges_[i].first)))
205 .At(cc->InputTimestamp()));
206 }
207 } else {
208 for (int i = 0; i < (int)ranges_.size(); ++i) {
209 auto output = absl::make_unique<std::vector<T>>();
210 output->insert(
211 output->end(),
212 std::make_move_iterator(input_vector->begin() + ranges_[i].first),
213 std::make_move_iterator(input_vector->begin() +
214 ranges_[i].second));
215 cc->Outputs().Index(i).Add(output.release(), cc->InputTimestamp());
216 }
217 }
218 }
219
220 return absl::OkStatus();
221 }
222
223 template <typename U, IsNotMovable<U> = true>
224 absl::Status ProcessMovableElements(CalculatorContext* cc) {
225 UNUSED(cc);
226 return absl::InternalError("Cannot move non-movable elements.");
227 }
228
229 private:
230 static absl::Status checkRangesDontOverlap(
231 const ::mediapipe::SplitVectorCalculatorOptions& options) {
232 for (int i = 0; i < options.ranges_size() - 1; ++i) {
233 for (int j = i + 1; j < options.ranges_size(); ++j) {
234 const auto& range_0 = options.ranges(i);
235 const auto& range_1 = options.ranges(j);
236 if ((range_0.begin() >= range_1.begin() &&
237 range_0.begin() < range_1.end()) ||
238 (range_1.begin() >= range_0.begin() &&
239 range_1.begin() < range_0.end())) {
240 return absl::InvalidArgumentError(
241 "Ranges must be non-overlapping when using combine_outputs "
242 "option.");
243 }
244 }
245 }
246 return absl::OkStatus();
247 }
248
249 std::vector<std::pair<int32_t, int32_t>> ranges_;
250 int32_t max_range_end_ = -1;
251 int32_t total_elements_ = 0;
252 bool element_only_ = false;
253 bool combine_outputs_ = false;
254};
