1// This file defines an API to define a node's ports in a concise, type-safe
2// way. Example usage in a node:
3//
4// static constexpr Input<int> kBase("IN");
5// static constexpr Output<float> kOut("OUT");
6// static constexpr SideInput<float>::Optional kDelta("DELTA");
7// static constexpr SideOutput<float> kForward("FORWARD");
8//
9// Pass a CalculatorContext to a port to access the inputs or outputs in the
10// context. For example:
11//
12// kBase(cc) yields an InputShardAccess<int>
13// kOut(cc) yields an OutputShardAccess<float>
14// kDelta(cc) yields an InputSidePacketAccess<float>
15// kForward(cc) yields an OutputSidePacketAccess<float>

245template <typename ValueT, typename X, class CC>
246class MultiplePortAccess {
247 public:
248 using AccessT = decltype(SinglePortAccess<ValueT>(std::declval<CC*>(),
249 std::declval<X*>()));
250
251 MultiplePortAccess(CC* cc, X* first, int count)
252 : cc_(cc), first_(first), count_(count) {}
253
254 // TODO: maybe this should be size(), like in a standard C++
255 // container?
256 int Count() { return count_; }
257 AccessT operator[](int pos) {
258 ABSL_CHECK_GE(pos, 0);
259 ABSL_CHECK_LT(pos, count_);
260 return SinglePortAccess<ValueT>(cc_, &first_[pos]);
261 }

263 class Iterator {
264 public:
265 using iterator_category = std::input_iterator_tag;
266 using value_type = AccessT;
267 using difference_type = std::ptrdiff_t;
268 using pointer = AccessT*;
269 using reference = AccessT; // allowed; see e.g. std::istreambuf_iterator
270
271 Iterator(CC* cc, X* p) : cc_(cc), p_(p) {}
272 Iterator& operator++() {
273 ++p_;
274 return *this;
275 }
276 Iterator operator++(int) {
277 Iterator res = *this;
278 ++(*this);
279 return res;
280 }
281 bool operator==(const Iterator& other) const { return p_ == other.p_; }
282 bool operator!=(const Iterator& other) const { return !(*this == other); }
283 AccessT operator*() const { return SinglePortAccess<ValueT>(cc_, p_); }
284
285 private:
286 CC* cc_;
287 X* p_;
288 };
289
290 Iterator begin() { return Iterator(cc_, first_); }
291 Iterator end() { return Iterator(cc_, first_ + count_); }
292
293 private:
294 CC* cc_;
295 X* first_;
296 int count_;
297};

341// This template is used to define a port. Nodes should use it through one
342// of the aliases below (Input, Output, SideInput, SideOutput).
343template <typename Base, typename ValueT, bool IsOptionalV = false,
344 bool IsMultipleV = false>
345class PortCommon : public Base {
346 public:
347 using value_t = ValueT;
348 static constexpr bool kOptional = IsOptionalV;
349 static constexpr bool kMultiple = IsMultipleV;
350
351 using Optional = PortCommon<Base, ValueT, true, IsMultipleV>;
352 using Multiple = PortCommon<Base, ValueT, IsOptionalV, true>;
353 using SideFallback = SideFallbackT<Base, ValueT, IsOptionalV, IsMultipleV>;
354
355 explicit constexpr PortCommon(absl::string_view tag)
356 : Base(tag, kTypeId<ValueT>, IsOptionalV, IsMultipleV) {}
357
358 template <std::size_t N>
359 explicit constexpr PortCommon(const char (&tag)[N])
360 : Base(N, tag, kTypeId<ValueT>, IsOptionalV, IsMultipleV) {}
361
362 using PayloadT = ActualPayloadT<ValueT>;
363
364 auto operator()(CalculatorContext* cc) const {
365 return internal::AccessPort<PayloadT>(
366 std::integral_constant<bool, IsMultipleV>{}, *this, cc);
367 }
368
369 auto operator()(CalculatorContract* cc) const {
370 return internal::AccessPort<PayloadT>(
371 std::integral_constant<bool, IsMultipleV>{}, *this, cc);
372 }
373
374 private:
375 absl::Status AddToContract(CalculatorContract* cc) const {
376 if (kMultiple) {
377 AddMultiple(cc);
378 } else {
379 auto& pt = internal::GetCollection(cc, *this).Tag(this->Tag());
380 internal::SetType<value_t>(cc, pt);
381 if (kOptional) {
382 pt.Optional();
383 }
384 }
385 return {};
386 }
387
388 void AddMultiple(CalculatorContract* cc) const {
389 auto& collection = internal::GetCollection(cc, *this);
390 int count = collection.NumEntries(this->Tag());
391 for (int i = 0; i < count; ++i) {
392 internal::SetType<value_t>(cc, collection.Get(this->Tag(), i));
393 }
394 }

397 friend class internal::Contract;
398 template <typename B, typename VT, bool, bool>
399 friend class mediapipe::api2::SideFallbackT;

415template <typename Base, typename ValueT, bool IsOptionalV, bool IsMultipleV>
416class SideFallbackT : public Base {
417 public:
418 using value_t = ValueT;
419 static constexpr bool kOptional = IsOptionalV;
420 static constexpr bool kMultiple = IsMultipleV;
421 using Optional = SideFallbackT<Base, ValueT, true, IsMultipleV>;
422 using PayloadT = ActualPayloadT<ValueT>;
423
424 const char* Tag() const { return stream_port.Tag(); }
425
426 auto operator()(CalculatorContract* cc) const {
427 bool is_stream = true;
428 auto& stream_collection = internal::GetCollection(cc, stream_port);
429 auto* packet_type = internal::GetOrNull(stream_collection, Tag(), 0);
430 if (packet_type == nullptr) {
431 auto& side_collection = internal::GetCollection(cc, side_port);
432 packet_type = internal::GetOrNull(side_collection, Tag(), 0);
433 is_stream = false;
434 }
435 return internal::SinglePortAccess<PayloadT>(cc, packet_type, is_stream);
436 }
437
438 auto operator()(CalculatorContext* cc) const {
439 auto& stream_collection = internal::GetCollection(cc, stream_port);
440 auto& side_collection = internal::GetCollection(cc, side_port);
441 return internal::SinglePortAccess<PayloadT>(
442 cc, internal::GetOrNull(stream_collection, Tag(), 0),
443 internal::GetOrNull(side_collection, Tag(), 0));
444 }
445
446 template <std::size_t N>
447 explicit constexpr SideFallbackT(const char (&tag)[N])
448 : Base(N, tag, kTypeId<ValueT>, IsOptionalV, IsMultipleV),
449 stream_port(tag),
450 side_port(tag) {}
451
452 protected:
453 absl::Status AddToContract(CalculatorContract* cc) const {
454 stream_port.AddToContract(cc);
455 side_port.AddToContract(cc);
456 int connected_count =
457 stream_port(cc).IsConnected() + side_port(cc).IsConnected();
458 if (connected_count > 1)
459 return absl::InvalidArgumentError(absl::StrCat(
460 Tag(),
461 " can be connected as a stream or as a side packet, but not both"));
462 if (!IsOptionalV && connected_count == 0)
463 return absl::InvalidArgumentError(
464 absl::StrCat(Tag(), " must be connected"));
465 return {};
466 }

468 using StreamPort = PortCommon<Base, ValueT, true, IsMultipleV>;
469 using SidePort = PortCommon<typename internal::SideBase<Base>::type, ValueT,
470 true, IsMultipleV>;
471 StreamPort stream_port;
472 SidePort side_port;

478// An OutputShardAccess is returned when accessing an output stream within a
479// CalculatorContext (e.g. kOut(cc)), and provides a type-safe interface to
480// OutputStreamShard. Like that class, this class will not be usually named in
481// calculator code, but used as a temporary object (e.g. kOut(cc).Send(...)).
482//
483// If not connected (!IsConnected()) SetNextTimestampBound is safe to call and
484// does nothing.
485// All the sub-classes that define Send should implement it to be safe to to
486// call if not connected and do nothing in such case.

494 }

496 if (output_) output_->SetNextTimestampBound(timestamp);
497 }
498
499 bool IsClosed() const { return output_ ? output_->IsClosed() : true; }
500 void Close() {
501 if (output_) output_->Close();
502 }
503
504 bool IsConnected() const { return output_ != nullptr; }
505
506 protected:
507 const CalculatorContext& context_;
508 OutputStreamShard* output_;

601template <typename T>
602class InputShardAccess : public Packet<T> {
603 public:
604 const PacketBase& packet() const& { return *this; }
605 // Since InputShardAccess is currently created as a temporary, this avoids
606 // easy mistakes with dangling references.
607 PacketBase packet() const&& { return *this; }
608
609 bool IsDone() const { return stream_->IsDone(); }
610 bool IsConnected() const { return stream_ != nullptr; }
611
612 PacketBase Header() const { return FromOldPacket(stream_->Header()); }
613
614 // "Consume" requires exclusive ownership of the packet's payload. In the
615 // current interim implementation, InputShardAccess creates a new reference to
616 // the payload (as a Packet<T> instead of a type-erased Packet), which means
617 // the conditions for Consume would never be satisfied. This helper class
618 // defines wrappers for the Consume methods in Packet which temporarily erase
619 // the reference held by the underlying InputStreamShard.
620 // Note that we cannot simply take over the reference when InputShardAccess is
621 // created, because it is currently created as a temporary and we might create
622 // more than one instance for the same stream.
623 template <class U = T,
624 class = std::enable_if_t<std::is_same<U, T>{},
625 decltype(&Packet<U>::Consume)>>
626 absl::StatusOr<std::unique_ptr<U>> Consume() {
627 return WrapConsumeCall(&Packet<T>::Consume);
628 }
629
630 template <class V, class U = T,
631 std::enable_if_t<internal::IsCompatibleType<V, U>{}, int> = 0>
632 absl::StatusOr<std::unique_ptr<V>> Consume() {
633 return WrapConsumeCall(&Packet<T>::template Consume<V>);
634 }
635
636 template <class... F>
637 auto ConsumeAndVisit(F&&... args) {
638 auto f = &Packet<T>::template ConsumeAndVisit<F...>;
639 return WrapConsumeCall(f, std::forward<F>(args)...);
640 }
641
642 private:
643 InputShardAccess(const CalculatorContext&, InputStreamShard* stream)
644 : Packet<T>(stream ? FromOldPacket(stream->Value()).template As<T>()
645 : Packet<T>()),
646 stream_(stream) {}
647
648 template <class F, class... A>
649 auto WrapConsumeCall(F f, A&&... args) {
650 stream_->Value() = {};
651 auto result = (this->*f)(std::forward<A>(args)...);
652 if (!result.ok()) {
653 stream_->Value() = ToOldPacket(*this);
654 }
655 return result;
656 }
657
658 InputStreamShard* stream_;
659
660 friend InputShardAccess<T> internal::SinglePortAccess<T>(
661 mediapipe::CalculatorContext*, InputStreamShard*);
662};
