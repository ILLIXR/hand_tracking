30class OutputStreamManager;

32// The output stream spec shared across all output stream shards and their
33// output stream manager.
34struct OutputStreamSpec {
35 // Triggers the error callback with absl::Status info when an error
36 // occurs.
37 void TriggerErrorCallback(const absl::Status& status) const {
38 ABSL_CHECK(error_callback);
39 error_callback(status);
40 }
41
42 std::string name;
43 const PacketType* packet_type;
44 std::function<void(absl::Status)> error_callback;
45 bool locked_intro_data;
46 // Those three variables are the intro data protected by locked_intro_data.
47 bool offset_enabled;
48 TimestampDiff offset;
49 Packet header;
50};
51
52// OutputStreamShard, a subclass of OutputStream, holds an output queue
53// and a timetstamp bound of an output stream. Each call to
54// Calculator::Open(), Calculator::Process(), and Calculator::Close() can only
55// access its own OutputStreamShard.
56class OutputStreamShard : public OutputStream {
57 public:
58 OutputStreamShard();
59
60 void SetSpec(OutputStreamSpec* output_stream_spec);
61
62 // TODO Remove this interface from OutputStream? No client is using
63 // this API.
64 const std::string& Name() const final;
65
66 // Sets the next timestamp bound in the OutputStreamShard.
67 void SetNextTimestampBound(Timestamp timestamp) final;
68 // Returns the next timestamp bound.
69 Timestamp NextTimestampBound() const final { return next_timestamp_bound_; }
70
71 // Marks the stream as closed in the OutputStreamShard. However, the output
72 // stream will still be open until the OutputStreamHandler processes the
73 // OutputStreamShard and executes OutputStreamManager::Close().
74 void Close() final;
75 bool IsClosed() const final;
76
77 // Sets the offset.
78 void SetOffset(TimestampDiff offset) final;
79
80 TimestampDiff Offset() const final { return output_stream_spec_->offset; }
81
82 bool OffsetEnabled() const final {
83 return output_stream_spec_->offset_enabled;
84 }
85
86 // Sets the stream header.
87 void SetHeader(const Packet& packet) final;
88 // Returns a const reference to the header packet.
89 const Packet& Header() const final;
90
91 // Adds a packet to the output stream shard.
92 void AddPacket(const Packet& packet) final;
93 // Takes an rvalue reference of the packet and moves the packet to the output
94 // stream shard.
95 void AddPacket(Packet&& packet) final;
96
97 // Returns true if the output queue is empty.
98 bool IsEmpty() const { return output_queue_.empty(); }
99 // Returns the timestamp of the last added packet in the output queue.
100 Timestamp LastAddedPacketTimestamp() const;
101
102 private:
103 // AddPacketInternal template is called by either AddPacket(Packet&& packet)
104 // or AddPacket(const Packet& packet).
105 template <typename T>
106 absl::Status AddPacketInternal(T&& packet);
107
108 // Returns a pointer to the output queue.
109 std::list<Packet>* OutputQueue() { return &output_queue_; }
110 const std::list<Packet>* OutputQueue() const { return &output_queue_; }
111
112 // Resets data members.
113 void Reset(Timestamp next_timestamp_bound, bool close);
114
115 // A pointer to the output stream spec object, which is owned by the output
116 // stream manager.
117 OutputStreamSpec* output_stream_spec_;
118 std::list<Packet> output_queue_;
119 bool closed_;
120 Timestamp next_timestamp_bound_;
121 // Equal to next_timestamp_bound_ only if the bound has been explicitly set
122 // by the calculator. This is needed for parallel Process() calls,
123 // in order to avoid propagating the initial next_timestamp_bound_, which
124 // does not reflect the output of Process() for preceding timestamps.
125 Timestamp updated_next_timestamp_bound_;
126
127 // Accesses OutputStreamShard for profiling.
128 friend class GraphProfiler;
129 // Accesses OutputStreamShard for profiling.
130 friend class GraphTracer;
131 // Accesses OutputStreamShard for profiling.
132 friend class PerfettoTraceScope;
133 // Accesses OutputStreamShard for post processing.
134 friend class OutputStreamManager;
135};
