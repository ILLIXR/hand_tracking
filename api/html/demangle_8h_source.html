18#ifndef MEDIAPIPE_HAS_CXA_DEMANGLE
19// We only support some compilers that support __cxa_demangle.
20// TODO: Checks if Android NDK has fixed this issue or not.
21#if defined(__ANDROID__) && (defined(__i386__) || defined(__x86_64__))
22#define MEDIAPIPE_HAS_CXA_DEMANGLE 0
23#elif (__GNUC__ >= 4 || (__GNUC__ >= 3 && __GNUC_MINOR__ >= 4)) && \
24 !defined(__mips__)
25#define MEDIAPIPE_HAS_CXA_DEMANGLE 1
26#elif defined(__clang__) && !defined(_MSC_VER)
27#define MEDIAPIPE_HAS_CXA_DEMANGLE 1
28#else
29#define MEDIAPIPE_HAS_CXA_DEMANGLE 0
30#endif
31#endif

42// Demangle a mangled symbol name and return the demangled name.
43// If 'mangled' isn't mangled in the first place, this function
44// simply returns 'mangled' as is.
45//
46// This function is used for demangling mangled symbol names such as
47// '_Z3bazifdPv'. It uses abi::__cxa_demangle() if your compiler has
48// the API. Otherwise, this function simply returns 'mangled' as is.
49//
50// Currently, we support only GCC 3.4.x or later for the following
51// reasons.
52//
53// - GCC 2.95.3 doesn't have cxxabi.h
54// - GCC 3.3.5 and ICC 9.0 have a bug. Their abi::__cxa_demangle()
55// returns junk values for non-mangled symbol names (ex. function
56// names in C linkage). For example,
57// abi::__cxa_demangle("main", 0, 0, &status)
58// returns "unsigned long" and the status code is 0 (successful).
59//
60// Also,
61//
62// - MIPS is not supported because abi::__cxa_demangle() is not defined.
63// - Android x86 is not supported because STLs don't define __cxa_demangle
64//
65// Prefer using MediaPipeTypeStringOrDemangled<T>() when possible (defined
66// in type_map.h).
67inline std::string Demangle(const char* mangled) {
68 int status = 0;
69 char* demangled = nullptr;
70#if MEDIAPIPE_HAS_CXA_DEMANGLE
71 demangled = abi::__cxa_demangle(mangled, nullptr, nullptr, &status);
72#endif
73 std::string out;
74 if (status == 0 && demangled != nullptr) { // Demangling succeeeded.
75 out.append(demangled);
76 free(demangled);
77 } else {
78 out.append(mangled);
79 }
80 return out;
81}
