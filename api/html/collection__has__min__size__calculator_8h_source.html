28// Deterimines if an input iterable collection has a minimum size, specified
29// in CollectionHasMinSizeCalculatorOptions. Example usage:
30// node {
31// calculator: "IntVectorHasMinSizeCalculator"
32// input_stream: "ITERABLE:input_int_vector"
33// output_stream: "has_min_ints"
34// options {
35// [mediapipe.CollectionHasMinSizeCalculatorOptions.ext] {
36// min_size: 2
37// }
38// }
39// }
40// Optionally, uses a side packet to override `min_size` specified in the
41// calculator options.
42template <typename IterableT>
43class CollectionHasMinSizeCalculator : public CalculatorBase {
44 public:
45 static absl::Status GetContract(CalculatorContract* cc) {
46 RET_CHECK(cc->Inputs().HasTag("ITERABLE"));
47 RET_CHECK_EQ(1, cc->Inputs().NumEntries());
48
49 RET_CHECK_EQ(1, cc->Outputs().NumEntries());
50
51 RET_CHECK_GE(
52 cc->Options<::mediapipe::CollectionHasMinSizeCalculatorOptions>()
53 .min_size(),
54 0);
55
56 cc->Inputs().Tag("ITERABLE").Set<IterableT>();
57 cc->Outputs().Index(0).Set<bool>();
58
59 // Optional input side packet that determines `min_size_`.
60 if (cc->InputSidePackets().NumEntries() > 0) {
61 cc->InputSidePackets().Index(0).Set<int>();
62 }
63 return absl::OkStatus();
64 }
65
66 absl::Status Open(CalculatorContext* cc) override {
67 cc->SetOffset(TimestampDiff(0));
68 min_size_ =
69 cc->Options<::mediapipe::CollectionHasMinSizeCalculatorOptions>()
70 .min_size();
71 // Override `min_size` if passed as side packet.
72 if (cc->InputSidePackets().NumEntries() > 0 &&
73 !cc->InputSidePackets().Index(0).IsEmpty()) {
74 min_size_ = cc->InputSidePackets().Index(0).Get<int>();
75 }
76 return absl::OkStatus();
77 }
78
79 absl::Status Process(CalculatorContext* cc) override {
80 const IterableT& input = cc->Inputs().Tag("ITERABLE").Get<IterableT>();
81 bool has_min_size = input.size() >= (size_t)min_size_;
82
83 cc->Outputs().Index(0).AddPacket(
84 MakePacket<bool>(has_min_size).At(cc->InputTimestamp()));
85
86 return absl::OkStatus();
87 }
88
89 private:
90 int min_size_ = 0;
91};
