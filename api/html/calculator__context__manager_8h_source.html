33namespace mediapipe {
34
35// Calculator context manager owns and manages all calculator context objects of
36// a calculator node.
37class CalculatorContextManager {
38 public:
39 CalculatorContextManager() {}
40
41 void Initialize(CalculatorState* calculator_state,
42 std::shared_ptr<tool::TagMap> input_tag_map,
43 std::shared_ptr<tool::TagMap> output_tag_map,
44 bool calculator_run_in_parallel);
45
46 // Sets the callback that can setup the input and output stream shards in a
47 // newly constructed calculator context. Then, initializes the default
48 // calculator context.
49 absl::Status PrepareForRun(
50 std::function<absl::Status(CalculatorContext*)> setup_shards_callback);
51
52 // Invoked by CalculatorNode::CleanupAfterRun().
53 void CleanupAfterRun() ABSL_LOCKS_EXCLUDED(contexts_mutex_);
54
55 // Returns true if the default calculator context has been initialized.
56 bool HasDefaultCalculatorContext() const {
57 return default_context_ != nullptr;
58 }
59
60 // Returns a pointer to the default calculator context that is used for
61 // sequential execution. A source node should always reuse its default
62 // calculator context.
63 CalculatorContext* GetDefaultCalculatorContext() const;
64
65 // Returns the context with the smallest input timestamp in active_contexts_.
66 // The input timestamp of the calculator context is returned in
67 // *context_input_timestamp.
68 CalculatorContext* GetFrontCalculatorContext(
69 Timestamp* context_input_timestamp) ABSL_LOCKS_EXCLUDED(contexts_mutex_);
70
71 // For sequential execution, returns a pointer to the default calculator
72 // context. For parallel execution, creates or reuses a calculator context,
73 // and inserts the calculator context with the given input timestamp into
74 // active_contexts_. Returns a pointer to the prepared calculator context.
75 // The ownership of the calculator context object isn't tranferred to the
76 // caller.
77 CalculatorContext* PrepareCalculatorContext(Timestamp input_timestamp)
78 ABSL_LOCKS_EXCLUDED(contexts_mutex_);
79
80 // Removes the context with the smallest input timestamp from active_contexts_
81 // and moves the calculator context to idle_contexts_. The caller must
82 // guarantee that the output shards in the calculator context have been
83 // propagated before calling this function.
84 void RecycleCalculatorContext() ABSL_LOCKS_EXCLUDED(contexts_mutex_);
85
86 // Returns true if active_contexts_ is non-empty.
87 bool HasActiveContexts() ABSL_LOCKS_EXCLUDED(contexts_mutex_);
88
89 int NumberOfContextTimestamps(
90 const CalculatorContext& calculator_context) const {
91 return calculator_context.NumberOfTimestamps();
92 }
93
94 bool ContextHasInputTimestamp(
95 const CalculatorContext& calculator_context) const {
96 return calculator_context.HasInputTimestamp();
97 }
98
99 void PushInputTimestampToContext(CalculatorContext* calculator_context,
100 Timestamp input_timestamp) {
101 ABSL_CHECK(calculator_context);
102 calculator_context->PushInputTimestamp(input_timestamp);
103 }
104
105 void PopInputTimestampFromContext(CalculatorContext* calculator_context) {
106 ABSL_CHECK(calculator_context);
107 calculator_context->PopInputTimestamp();
108 }
109
110 void SetGraphStatusInContext(CalculatorContext* calculator_context,
111 const absl::Status& status) {
112 ABSL_CHECK(calculator_context);
113 calculator_context->SetGraphStatus(status);
114 }
115
116 private:
117 CalculatorState* calculator_state_;
118 std::shared_ptr<tool::TagMap> input_tag_map_;
119 std::shared_ptr<tool::TagMap> output_tag_map_;
120 bool calculator_run_in_parallel_;
121
122 // The callback to setup the input and output stream shards in a newly
123 // constructed calculator context.
124 // NOTE: This callback invokes input/output stream handler methods.
125 // The callback is used to break the circular dependency between
126 // calculator context manager and input/output stream handlers.
127 std::function<absl::Status(CalculatorContext*)> setup_shards_callback_;
128
129 // The default calculator context that is always reused for sequential
130 // execution. It is also used by Open() and Close() method of a parallel
131 // calculator.
132 std::unique_ptr<CalculatorContext> default_context_;
133 // The mutex for synchronizing the operations on active_contexts_ and
134 // idle_contexts_ during parallel execution.
135 absl::Mutex contexts_mutex_;
136 // A map from input timestamps to calculator contexts.
137 std::map<Timestamp, std::unique_ptr<CalculatorContext>> active_contexts_
138 ABSL_GUARDED_BY(contexts_mutex_);
139 // Idle calculator contexts that are ready for reuse.
140 std::deque<std::unique_ptr<CalculatorContext>> idle_contexts_
141 ABSL_GUARDED_BY(contexts_mutex_);
142};
143
144} // namespace mediapipe
