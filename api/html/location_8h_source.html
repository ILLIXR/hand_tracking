15// A container for location data, representing location information in an image.
16// This wrapper provides two functionalities:
17// 1. Factory methods for creation of Location objects and thus LocationData
18// protocol buffers. These methods guarantee a valid location data and are
19// the prefer way of creating such.
20// 2. Accessors which allow for extracting location information in various
21// formats. If necessary, the location information is converted.
22

37namespace mediapipe {
38
39class Location {
40 public:
41 // CREATION METHODS.
42 Location();
43 // Constructs a location wrapping the specified location data. Checks the
44 // validity of the input and crashes upon failure.
45 explicit Location(const LocationData& location_data);
46 // Creates a location of type GLOBAL, i.e. a location representing the full
47 // image.
48 static Location CreateGlobalLocation();
49 // Creates a location of type BOUNDING_BOX, i.e. it is based on a bounding box
50 // defined by its upper left corner (xmin, ymin) and its width and height.
51 static Location CreateBBoxLocation(int xmin, int ymin, int width, int height);
52 // Creates a location of type BOUNDING_BOX from bounding boxes in various
53 // formats.
54 static Location CreateBBoxLocation(const Rectangle_i& rect);
55 static Location CreateBBoxLocation(const ::mediapipe::BoundingBox& bbox);
56 // Creates a location of type RELATIVE_BOUNDING_BOX, i.e. it is based on a
57 // bounding box defined by its upper left corner (xmin, ymin) and its width
58 // and height, all relative to the image dimensions.
59 static Location CreateRelativeBBoxLocation(float relative_xmin,
60 float relative_ymin,
61 float relative_width,
62 float relative_height);
63 // Creates a location of type RELATIVE_BOUNDING_BOX from bounding boxes in
64 // various formats.
65 static Location CreateRelativeBBoxLocation(const Rectangle_f& relative_rect);
66
67 // Returns the location type describing the type of data it contains. This
68 // type is set at creation time based on the one of the above factory methods.
69 LocationData::Format GetFormat() const;
70 // Checks the validity of the specified location_data, i.e. whether all the
71 // necessary fields for the location data type are set.
72 static bool IsValidLocationData(const LocationData& location_data);
73
74 // MODIFIERS
75 // Scales the current bounding box (x,y,width,height) by the "scale" amount
76 // (1.0f means no scale, <1.0f means scale down, and >1.0f means scale up).
77 // Returns *this.
78 //
79 // NOTE: it does not handle masks.
80 Location& Scale(float scale);
81
82 // Resizes the location such that it is the tighest square location containing
83 // centered the original location. It supports locations of type GLOBAL,
84 // BOUNDING_BOX and RELATIVE_BOUNDING_BOX, otherwise it CHECK-fails. The user
85 // must specify the image dimensions. Returns *this.
86 Location& Square(int image_width, int image_height);
87
88 // If the location is larger than the image, then in some cases it is
89 // beneficial to translate the location such that the image is centered
90 // within the location. The following method achieves this translation.
91 // Returns *this.
92 Location& ShiftToFitBestIntoImage(int image_width, int image_height);
93
94 // Replaces the location with the intersection of the current location the
95 // specified crop rectangle. Locations of type GLOBAL remain
96 // unmodified. The image size is used for locations of type
97 // BOUNDARY_BOX and MASK. Using this override for locations of type
98 // RELATIVE_BOUNDING_BOX is an error. Use the Rectangle_f override of this
99 // function instead.
100 // This operation is useful when one needs to make sure that the location is
101 // fully contained within the specified image. Returns *this.
102 Location& Crop(const Rectangle_i& crop_rectangle);
103
104 // Replaces the location with the intersection of the current location and the
105 // specified crop rectangle. Locations of type GLOBAL remain
106 // unmodified. This override is only for locations of type
107 // RELATIVE_BOUNDING_BOX. Use the Rectangle_i override of this
108 // function for other types of locations.
109 // This operation is useful when one needs to make sure that the location is
110 // fully contained within the specified image. Returns *this.
111 Location& Crop(const Rectangle_f& crop_rectangle);
112
113 // ACCESSORS.
114 // Non-type converting accessor: returns the requested data only if the output
115 // format is consistent with the location data type. E.g. if one requests a
116 // rectangle, then the wrapped location data should be of type BOUNDING_BOX.
117 // Accessor for location data type BOUNDING_BOX with two possible return types
118 // Rectangle_i and mediapipe.::mediapipe::BoundingBox.
119 template <typename T>
120 T GetBBox() const;
121 // Accessor for location data type RELATIVE_BOUNDING_BOX.
122 Rectangle_f GetRelativeBBox() const;
123
124 // Accessor for relative_keypoints in location data. Relative keypoints are
125 // specified with x and y coordinates, where both x and y are relative to the
126 // image width and height, respectively, and are in the range [0, 1]. Fails if
127 // location data is not of type RELATIVE_BOUNDING_BOX.
128 std::vector<Point2_f> GetRelativeKeypoints() const;
129
130 // Type converting accessor: returns the requested data in the specified
131 // output type. If the location data is in a format not directly convertible
132 // to the specified return type the following conversion principles are used:
133 // - Rectangle -> Mask: the rectangle is converted to a mask with all
134 // pixels inside the rectangle being foreground pixels.
135 // - Mask -> Rectangle: the tightest enclosing rectangle to the mask is the
136 // rectangle representation of a mask.
137 // - Global -> Rectangle, Mask: all the pixels in the image are considered
138 // foreground. Thus, the equivalent rectangle and mask are a rectangle
139 // covering the full image.
140 // The supported output types are the same as for GetBBox() above: Rectangle_i
141 // and mediapipe.::mediapipe::BoundingBox.
142 template <typename T>
143 T ConvertToBBox(int image_width, int image_height) const;
144 Rectangle_f ConvertToRelativeBBox(int image_width, int image_height) const;
145 // Returns keypoints in absolute pixel coordinates.
146 std::vector<Point2_i> ConvertToKeypoints(int image_width,
147 int image_height) const;
148
149 // Sets the keypoints.
150 void SetRelativeKeypoints(const std::vector<Point2_f>& keypoints);
151
152 // Serializes and deserializes the location object.
153 void ConvertToProto(LocationData* proto) const;
154 LocationData ConvertToProto() const;
155 void SetFromProto(const LocationData& proto);
156
157#if !defined(MEDIAPIPE_MOBILE) && !defined(MEDIAPIPE_LITE)
158 // Deep equality comparison.
159 bool operator==(const Location& other) const;
160 bool operator!=(const Location& other) const;
161#endif // !defined(MEDIAPIPE_MOBILE) && !defined(MEDIAPIPE_LITE)
162
163 private:
164 // The wrapped location data.
165 LocationData location_data_;
166};
167} // namespace mediapipe

169#endif // MEDIAPIPE_FRAMEWORK_FORMATS_LOCATION_H_
