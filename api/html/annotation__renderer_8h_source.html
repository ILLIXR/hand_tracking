26// The renderer library for rendering data on images.
27//
28// Example usage:
29//
30// AnnotationRenderer renderer;
31//
32// std::unique_ptr<cv::Mat> mat_image(new cv::Mat(kImageHeight, kImageWidth,
33// CV_8UC3));
34//
35// renderer.AdoptImage(mat_image.get());
36//
37// RenderData render_data_0;
38// <FILL RENDER_DATA_0 WITH ANNOTATIONS>
39//
40// renderer.RenderDataOnImage(render_data_0);
41//
42// RenderData render_data_1;
43// <FILL RENDER_DATA_1 WITH ANNOTATIONS>
44//
45// renderer.RenderDataOnImage(render_data_1);
46//
47// UseRenderedImage(mat_image.get());
48class AnnotationRenderer {
49 public:
50 explicit AnnotationRenderer() {}
51
52 explicit AnnotationRenderer(const cv::Mat& mat_image)
53 : image_width_(mat_image.cols),
54 image_height_(mat_image.rows),
55 mat_image_(mat_image.clone()) {}
56
57 // Renders the image with the input render data.
58 void RenderDataOnImage(const RenderData& render_data);
59
60 // Resets the renderer with a new image. Does not own input_image. input_image
61 // must not be modified by caller during rendering.
62 void AdoptImage(cv::Mat* input_image);
63
64 // Gets image dimensions.
65 int GetImageWidth() const;
66 int GetImageHeight() const;
67
68 // Sets whether text should be rendered upside down. This is default to false
69 // and text is rendered assuming the underlying image has its origin at the
70 // top-left corner. Set it to true if the image origin is at the bottom-left
71 // corner.
72 void SetFlipTextVertically(bool flip);
73
74 // For GPU rendering optimization in AnnotationOverlayCalculator.
75 // Scale all incoming coordinates,sizes,thickness,etc. by this amount.
76 // Should be in the range (0-1].
77 // See 'gpu_scale_factor' in annotation_overlay_calculator.proto
78 void SetScaleFactor(float scale_factor);
79 float GetScaleFactor() { return scale_factor_; }
80
81 private:
82 // Draws a rectangle on the image as described in the annotation.
83 void DrawRectangle(const RenderAnnotation& annotation);
84
85 // Draws a filled rectangle on the image as described in the annotation.
86 void DrawFilledRectangle(const RenderAnnotation& annotation);
87
88 // Draws an oval on the image as described in the annotation.
89 void DrawOval(const RenderAnnotation& annotation);
90
91 // Draws a filled oval on the image as described in the annotation.
92 void DrawFilledOval(const RenderAnnotation& annotation);
93
94 // Draws an arrow on the image as described in the annotation.
95 void DrawArrow(const RenderAnnotation& annotation);
96
97 // Draws a point on the image as described in the annotation.
98 void DrawPoint(const RenderAnnotation& annotation);
99 void DrawPoint(const RenderAnnotation::Point& point,
100 const RenderAnnotation& annotation);
101
102 // Draws scribbles on the image as described in the annotation.
103 void DrawScribble(const RenderAnnotation& annotation);
104
105 // Draws a line segment on the image as described in the annotation.
106 void DrawLine(const RenderAnnotation& annotation);
107
108 // Draws a 2-tone line segment on the image as described in the annotation.
109 void DrawGradientLine(const RenderAnnotation& annotation);
110
111 // Draws a text on the image as described in the annotation.
112 void DrawText(const RenderAnnotation& annotation);
113
114 // Draws a rounded rectangle on the image as described in the annotation.
115 void DrawRoundedRectangle(const RenderAnnotation& annotation);
116
117 // Draws a filled rounded rectangle on the image as described in the
118 // annotation.
119 void DrawFilledRoundedRectangle(const RenderAnnotation& annotation);
120
121 // Helper function for drawing a rectangle with rounded corners. The
122 // parameters are the same as in the OpenCV function rectangle().
123 // corner_radius: A positive int value defining the radius of the round
124 // corners.
125 void DrawRoundedRectangle(cv::Mat src, cv::Point top_left,
126 cv::Point bottom_right,
127 const cv::Scalar& line_color, int thickness = 1,
128 int line_type = 8, int corner_radius = 0);
129
130 // Computes the font scale from font_face, size and thickness.
131 double ComputeFontScale(int font_face, int font_size, int thickness);
132
133 // Width and Height of the image (in pixels).
134 int image_width_ = -1;
135 int image_height_ = -1;
136
137 // The image for rendering.
138 cv::Mat mat_image_;
139
140 // See SetFlipTextVertically(bool).
141 bool flip_text_vertically_ = false;
142
143 // See SetScaleFactor(float)
144 float scale_factor_ = 1.0;
145};
146} // namespace mediapipe
147
148#endif // MEDIAPIPE_UTIL_ANNOTATION_RENDERER_H_
