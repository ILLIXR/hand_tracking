37// A CalculatorContext provides information about the graph it is running
38// inside of through a number of accessor functions: Inputs(), Outputs(),
39// InputSidePackets(), Options(), etc.
40//
41// CalculatorBase APIs, such as CalculatorBase::Open(CalculatorContext* cc),
42// CalculatorBase::Process(CalculatorContext* cc), and
43// CalculatorBase::Close(CalculatorContext* cc), will only interact with
44// its own CalculatorContext object for exchanging data with the framework.
45class CalculatorContext {
46 public:
47 CalculatorContext(CalculatorState* calculator_state,
48 std::shared_ptr<tool::TagMap> input_tag_map,
49 std::shared_ptr<tool::TagMap> output_tag_map)
50 : calculator_state_(calculator_state),
51 inputs_(std::move(input_tag_map)),
52 outputs_(std::move(output_tag_map)) {}
53
54 CalculatorContext(const CalculatorContext&) = delete;
55 CalculatorContext& operator=(const CalculatorContext&) = delete;
56
57 const std::string& NodeName() const;
58 int NodeId() const;
59 const std::string& CalculatorType() const;
60 // Returns the options given to this calculator. The Calculator or
61 // CalculatorBase implementation may get its options by calling
62 // GetExtension() on the result.
63 const CalculatorOptions& Options() const;
64
65 // Returns the options given to this calculator. Template argument T must
66 // be the type of the protobuf extension message or the protobuf::Any
67 // message containing the options.
68 template <class T>
69 const T& Options() const {
70 return calculator_state_->Options<T>();
71 }
72
73 template <class T>
74 bool HasOptions() const {
75 return calculator_state_->HasOptions<T>();
76 }
77
78 // Returns a counter using the graph's counter factory. The counter's name is
79 // the passed-in name, prefixed by the calculator node's name (if present) or
80 // the calculator's type (if not).
81 Counter* GetCounter(const std::string& name);
82
83 // Returns the counter set, which can be used to create new counters.
84 // No prefix is added to counters created in this way.
85 CounterFactory* GetCounterFactory();
86
87 // Returns the current input timestamp, or Timestamp::Unset if there are
88 // no input packets.
89 Timestamp InputTimestamp() const {
90 return input_timestamps_.empty() ? Timestamp::Unset()
91 : input_timestamps_.front();
92 }
93
94 // Returns a reference to the input side packet set.
95 const PacketSet& InputSidePackets() const;
96 // Returns a reference to the output side packet collection.
97 OutputSidePacketSet& OutputSidePackets();
98 // Returns a reference to the input stream collection.
99 // You may consume or move the value packets from the Inputs.
100 InputStreamShardSet& Inputs();
101 // Returns a const reference to the input stream collection.
102 const InputStreamShardSet& Inputs() const;
103 // Returns a reference to the output stream collection.
104 OutputStreamShardSet& Outputs();
105 // Returns a const reference to the output stream collection.
106 const OutputStreamShardSet& Outputs() const;
107
108 // Sets this packet timestamp offset for Packets going to all outputs.
109 // If you only want to set the offset for a single output stream then
110 // use OutputStream::SetOffset() directly.
111 void SetOffset(TimestampDiff offset);
112
113 // DEPRECATED: This was intended to get graph run status during
114 // `CalculatorBase::Close` call. However, `Close` can run simultaneously with
115 // other calculators `CalculatorBase::Process`, hence the actual graph
116 // status may change any time and returned graph status here does not
117 // necessarily reflect the actual graph status.
118 //
119 // As an alternative, instead of checking graph status in `Close` and doing
120 // work for "done" state, you can enable timestamp bound processing for your
121 // calculator (`CalculatorContract::SetProcessTimestampBounds`) to trigger
122 // `Process` on timestamp bound updates and handle "done" state there.
123 // Check examples in:
124 // mediapipe/framework/calculator_graph_summary_packet_test.cc.
125 //
126 ABSL_DEPRECATED("Does not reflect the actual graph status.")
127 absl::Status GraphStatus() const { return graph_status_; }
128
129 ProfilingContext* GetProfilingContext() const {
130 return calculator_state_->GetSharedProfilingContext().get();
131 }
132
133 template <typename T>
134 ServiceBinding<T> Service(const GraphService<T>& service) {
135 return ServiceBinding<T>(calculator_state_->GetServiceObject(service));
136 }
137
138 private:
139 int NumberOfTimestamps() const {
140 return static_cast<int>(input_timestamps_.size());
141 }
142
143 bool HasInputTimestamp() const { return !input_timestamps_.empty(); }
144
145 // Adds a new input timestamp by the friend class CalculatorContextManager.
146 void PushInputTimestamp(Timestamp input_timestamp) {
147 input_timestamps_.push(input_timestamp);
148 }
149
150 void PopInputTimestamp() {
151 ABSL_CHECK(!input_timestamps_.empty());
152 input_timestamps_.pop();
153 }
154
155 void SetGraphStatus(const absl::Status& status) { graph_status_ = status; }
156
157 // Interface for the friend class Calculator.
158 const InputStreamSet& InputStreams() const;
159 const OutputStreamSet& OutputStreams() const;
160
161 // Stores the shared data across all CalculatorContext objects, including
162 // input side packets, calculator options, node name, etc.
163 // TODO: Removes unnecessary fields from CalculatorState after
164 // migrating all clients to CalculatorContext.
165 CalculatorState* calculator_state_;
166 InputStreamShardSet inputs_;
167 OutputStreamShardSet outputs_;
168 // Created on-demand when needed by legacy APIs. No synchronization needed
169 // because all possible callers are already serialized.
170 mutable std::unique_ptr<InputStreamSet> input_streams_;
171 mutable std::unique_ptr<OutputStreamSet> output_streams_;
172 // The queue of timestamp values to Process() in this calculator context.
173 std::queue<Timestamp> input_timestamps_;
174
175 // The status of the graph run. Only used when Close() is called.
176 absl::Status graph_status_;
177
178 // Accesses CalculatorContext for setting input timestamp.
179 friend class CalculatorContextManager;
180};
