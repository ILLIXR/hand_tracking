38// Holds the information needed for tag/index retrieval for stream and
39// side packet lists.
40class TagMap {
41 public:
42 // Struct to hold the initial id and the number of indexes per tag.
43 struct TagData {
44 TagData() : id(-1), count(0) {}
45 TagData(CollectionItemId first_id, int the_count)
46 : id(first_id), count(the_count) {}
47
48 // The initial id for this tag.
49 CollectionItemId id;
50 // The number of entries with this tag.
51 int count;
52 };
53
54 // Create a TagMap from a repeated string proto field of TAG:<index>:name.
55 // This is the most common usage:
56 // MP_ASSIGN_OR_RETURN(std::shared_ptr<TagMap> tag_map,
57 // tool::TagMap::Create(node.input_streams()));
58 static absl::StatusOr<std::shared_ptr<TagMap>> Create(
59 const proto_ns::RepeatedPtrField<ProtoString>& tag_index_names) {
60 std::shared_ptr<TagMap> output(new TagMap());
61 MP_RETURN_IF_ERROR(output->Initialize(tag_index_names));
62 return output;
63 }
64
65 // Create a TagMap from a TagAndNameInfo.
66 // TODO: Migrate callers and delete this method.
67 ABSL_DEPRECATED(
68 "Use mediapipe::tool::TagMap::Create(tag_index_names) instead.")
69 static absl::StatusOr<std::shared_ptr<TagMap>> Create(
70 const TagAndNameInfo& info) {
71 std::shared_ptr<TagMap> output(new TagMap());
72 MP_RETURN_IF_ERROR(output->Initialize(info));
73 return output;
74 }
75
76 // Returns a reference to the mapping from tag to tag data.
77 const absl::btree_map<std::string, TagData>& Mapping() const {
78 return mapping_;
79 }
80
81 // Returns the vector of names (indexed by CollectionItemId).
82 const std::vector<std::string>& Names() const { return names_; }
83
84 // Returns true if "this" and "other" use equivalent tags and indexes
85 // (disregards stream/side packet names).
86 bool SameAs(const TagMap& other) const;
87
88 // Returns canonicalized strings describing the TagMap.
89 proto_ns::RepeatedPtrField<ProtoString> CanonicalEntries() const;
90 // Returns a string description for debug purposes.
91 std::string DebugString() const;
92 // Returns a shorter description for debug purposes (doesn't include
93 // stream/side packet names).
94 std::string ShortDebugString() const;
95
96 // The following functions are directly utilized by collection.h see
97 // that file for comments.
98 bool HasTag(absl::string_view tag) const;
99 int NumEntries() const { return num_entries_; }
100 int NumEntries(absl::string_view tag) const;
101 CollectionItemId GetId(absl::string_view tag, int index) const;
102 std::set<std::string> GetTags() const;
103 std::pair<std::string, int> TagAndIndexFromId(CollectionItemId id) const;
104 CollectionItemId BeginId() const { return CollectionItemId(0); }
105 CollectionItemId EndId() const { return CollectionItemId(num_entries_); }
106 CollectionItemId BeginId(absl::string_view tag) const;
107 CollectionItemId EndId(absl::string_view tag) const;
108
109 private:
110 // Use static factory function TagMap::Create().
111 TagMap() {}
112
113 // Initialize the TagMap. Due to only having a factory function for
114 // creation, there is no way for a user to have an uninitialized TagMap.
115 absl::Status Initialize(
116 const proto_ns::RepeatedPtrField<ProtoString>& tag_index_names);
117
118 // Initialize from a TagAndNameInfo.
119 ABSL_DEPRECATED("Use Initialize(tag_index_names) instead.")
120 absl::Status Initialize(const TagAndNameInfo& info);
121
122 // Initialize names_ using a map from tag to the names for that tag.
123 void InitializeNames(
124 const std::map<std::string, std::vector<std::string>>& tag_to_names);
125
126 // The total number of entries under all tags.
127 int num_entries_;
128 // Mapping from tag to tag data.
129 absl::btree_map<std::string, TagData> mapping_;
130 // The names of the data (indexed by CollectionItemId).
131 std::vector<std::string> names_;
132};
133
134// Equal TagData structs define equal id ranges.
135inline bool operator==(const TagMap::TagData& d1, const TagMap::TagData& d2) {
136 return d1.id == d2.id && d1.count == d2.count;
137}
