36struct ABSL_DEPRECATED(
37 "Prefer using mediapipe::tool::TagMap instead, since this structure does "
38 "not "
39 "consider the TAG:INDEX:name notation.") TagAndNameInfo {
40 // The tag names. If this is empty then the collection will use
41 // indexes.
42 std::vector<std::string> tags;
43 // Names of the streams or side packets. If tags are used then
44 // tags.size() and names.size() must match.
45 std::vector<std::string> names;
46};

48// Create a TagAndNameInfo from a list of strings in the form TAG:name.
49// The TAG (including colon) is optional, but the entire list must either
50// always include tags or never include tags.
51ABSL_DEPRECATED(
52 "Prefer using mediapipe::tool::TagMap instead, since this method does not "
53 "support the TAG:INDEX:name notation. You can use Create() to create the "
54 "tag map, and then Names(), Mapping(), and other methods to access the "
55 "tag, index and name information.")
56absl::Status GetTagAndNameInfo(
57 const proto_ns::RepeatedPtrField<ProtoString>& tags_and_names,
58 TagAndNameInfo* info);

60// Create the proto field names in the form TAG:name based on a

70// The string is a valid name for an input stream, output stream,
71// side packet, and input collection. Names use only lower case letters,
72// numbers, and underscores.
73//
74// The reason for this restriction is threefold.
75// (1) To enforce a consistent style in graph configs.
76// (2) To distinguish between "arguments" to calculators and
77// trainer/calculator names.
78// (3) Because input side packet names end up in model directory names,
79// where lower case naming is the norm.
80absl::Status ValidateName(const std::string& name);
81// The string is a valid tag name. Tags use only upper case letters,
82// numbers, and underscores.
83absl::Status ValidateTag(const std::string& tag);
84
85// Parse a "Tag and Name" string into a tag and a name.
86// The format is an optional tag and colon, followed by a name.
87// Example 1: "VIDEO:frames2" -> tag: "VIDEO", name: "frames2"
88// Example 2: "video_frames_1" -> tag: "", name: "video_frames_1"
89absl::Status ParseTagAndName(absl::string_view tag_and_name, std::string* tag,
90 std::string* name);
91
92// Parse a generic TAG:index:name string. The format is a tag, then an
93// index, then a name. The tag and index are optional. If the index
94// is included, then the tag must be included. If no tag is used then
95// index is set to -1 (and should be assigned by argument position).
96// Examples:
97// "VIDEO:frames2" -> tag: "VIDEO", index: 0, name: "frames2"
98// "VIDEO:1:frames" -> tag: "VIDEO", index: 1, name: "frames"
99// "raw_frames" -> tag: "", index: -1, name: "raw_frames"
100absl::Status ParseTagIndexName(const std::string& tag_and_name,
101 std::string* tag, int* index, std::string* name);
102
103// Parse a generic TAG:index string. The format is a tag, then an index
104// with both being optional. If the tag is missing it is assumed to be
105// "" and if the index is missing then it is assumed to be 0. If the
106// index is provided then a colon (':') must be used.
107// Examples:
108// "VIDEO" -> tag: "VIDEO", index: 0
109// "VIDEO:1" -> tag: "VIDEO", index: 1
110// ":2" -> tag: "", index: 2
111// "" -> tag: "", index: 0
112absl::Status ParseTagIndex(const std::string& tag_and_index, std::string* tag,
113 int* index);
