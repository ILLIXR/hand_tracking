23// MonotonicClock is an interface for a Clock that never goes backward.
24// Successive returned values from Now() are guaranteed to be monotonically
25// non-decreasing, although they may not be monotonic with respect to values
26// returned from other instances of MonotonicClock.
27//
28// You can wrap any Clock object in a MonotonicClock using the
29// CreateMonotonicClock() factory method, including Clock::RealClock().
30// However, if you want a monotonic version of real time, it is strongly
31// recommended that you use the CreateSynchronizedMonotonicClock() factory
32// method, which wraps Clock::RealClock() and guarantees that values returned
33// from Now() are monotonic ACROSS instances of the class that are created by
34// CreateSynchronizedMonotonicClock().
35//
36// All methods support concurrent access.
37class MonotonicClock : public Clock {
38 public:
39 // The MonotonicClock state, which may be shared between MonotonicClocks.
40 struct State;
41
42 ~MonotonicClock() override {}
43
44 // The Clock interface (see util/time/clock.h).
45 //
46 // Return a monotonically non-decreasing time.
47 absl::Time TimeNow() override = 0;
48 // Sleep and SleepUntil guarantee only that the caller will sleep for at
49 // least as long as specified in monotonic time. The caller may sleep for
50 // much longer (in monotonic time) if monotonic time jumps far into the
51 // future. Whether or not this happens depends on the behavior of the raw
52 // clock.
53 void Sleep(absl::Duration d) override = 0;
54 void SleepUntil(absl::Time wakeup_time) override = 0;
55
56 // Get metrics about time corrections.
57 virtual void GetCorrectionMetrics(int* correction_count,
58 double* max_correction) = 0;
59 // Reset values returned by GetCorrectionMetrics().
60 virtual void ResetCorrectionMetrics() = 0;
61
62 // Factory methods.
63 //
64 // Create a MonotonicClock based on the given raw_clock. This clock will
65 // return monotonically non-decreasing values from Now(), but may not behave
66 // monotonically with respect to other instances created by this function,
67 // even if they are based on the same raw_clock. Caller owns raw_clock.
68 static MonotonicClock* CreateMonotonicClock(Clock* raw_clock);
69
70 // Create an instance of MonotonicClock that is based on Clock::RealClock().
71 // All such instance are synced with each other such that return values from
72 // Now() are monotonic across instances. This allows independently developed
73 // code bases to have private instances of the synchronized MonotonicClock
74 // and know that they will never see time anomalies when calling from one
75 // code base to another. Each instance can have its own correction callback.
76 // Unlike Clock::RealClock(), caller owns this object and should delete it
77 // when no longer needed.
78 static MonotonicClock* CreateSynchronizedMonotonicClock();
79};
